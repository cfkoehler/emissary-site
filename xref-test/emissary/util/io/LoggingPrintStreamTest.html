127         final Logger logger = (Logger) LoggerFactory.getLogger(LoggingPrintStreamTest.class.getName() + ".testMdcContextMap");

129         try (ByteArrayOutputStream baos = new ByteArrayOutputStream();

131                         new LoggingPrintStream(NullOutputStream.NULL_OUTPUT_STREAM, "STDTEST", logger, slf4jLevel, 30, TimeUnit.SECONDS)) {
132             final ConsoleAppender<ILoggingEvent> consoleAppender = new ConsoleAppender<>();
133             final PatternLayoutEncoder patternLayoutEncoder = new PatternLayoutEncoder();
134 
135             patternLayoutEncoder.setPattern("%5p %c - %X{MDC_KEY1} %X{MDC_KEY2} - %m%n");
136             patternLayoutEncoder.setContext(logger.getLoggerContext());
137             patternLayoutEncoder.start();
138 
139             consoleAppender.setEncoder(patternLayoutEncoder);
140             consoleAppender.setContext(logger.getLoggerContext());
141             consoleAppender.start();
142             consoleAppender.setOutputStream(baos);
143 
144             logger.addAppender(consoleAppender);
145             logger.setLevel(Level.INFO);
146             logger.setAdditive(false);
147 
148             MDC.put("MDC_KEY1", "MDC_VALUE1");
149             MDC.put("MDC_KEY2", "MDC_VALUE2");
150 
151             loggingPrintStream.println("MDC_TEST_MESSAGE");
152             loggingPrintStream.close();
153 
154             logger.detachAndStopAllAppenders();
155 
156             final String logMessage = baos.toString();
157 
158             assertEquals(" INFO emissary.util.io.LoggingPrintStreamTest.testMdcContextMap - MDC_VALUE1 MDC_VALUE2 - STDTEST : MDC_TEST_MESSAGE\n",

233     @Test
234     void testMultiThread() throws Exception {
235         final org.slf4j.event.Level slf4jLevel = org.slf4j.event.Level.INFO;
236 
237         try (LogbackTester logbackTester = new LogbackTester(LoggingPrintStreamTest.class.getName());
238                 ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();
239                 PrintStream printStream = new PrintStream(byteArrayOutputStream);
240                 LoggingPrintStream loggingPrintStream =
241                         new LoggingPrintStream(printStream, logbackTester.name, logbackTester.logger, slf4jLevel, 30, TimeUnit.SECONDS)) {
242             final int iterations = 25;
243             final int numberOfThreads = 5;
244 
245             final ExecutorService executorService = Executors.newFixedThreadPool(numberOfThreads);
246 
247             for (int i = 0; i < numberOfThreads; i++) {
248                 final String message = "Message from instance " + i;
249                 final Exception exception = new Exception("Exception from instance " + i);
250 
251                 executorService.submit(() -> {
252                     for (int j = 0; j < iterations; j++) {
253                         loggingPrintStream.println(message);
254                         exception.printStackTrace(loggingPrintStream);
255                     }
256                 });
257             }
258 
259             executorService.shutdown();
260             executorService.awaitTermination(10, TimeUnit.SECONDS);
261 
262             loggingPrintStream.close();
263 
264             assertEquals(iterations * numberOfThreads * 2, logbackTester.appender.list.size(), "Wrong number of log messages!");
265         }

268     public static class LogbackTester implements Closeable {
269         private static final AtomicInteger INSTANCE = new AtomicInteger(0);
270 
271         public final String name;
272         public final Logger logger;
273         public final ListAppender<ILoggingEvent> appender;
274 
275         public LogbackTester(final String prefix) {
276             Validate.notNull(prefix, "Required: prefix != null");
277 
278             name = prefix + "_" + INSTANCE.incrementAndGet();
279             logger = (Logger) LoggerFactory.getLogger(name);
280             appender = new ListAppender<>();
281 
282             appender.setContext(logger.getLoggerContext());
283             appender.start();
284             logger.addAppender(appender);
285             logger.setAdditive(false);
286         }
287 
288         public void checkLogList(final Level[] levels, final String[] messages, final boolean[] throwables) {
289             Validate.notNull(levels, "Required: levels != null");
290             Validate.notNull(messages, "Required: messages != null");
291             Validate.notNull(throwables, "Required: throwables != null");
292             Validate.isTrue(levels.length == messages.length, "Required: levels.length == messages.length");
293             Validate.isTrue(levels.length == throwables.length, "Required: levels.length == throwables.length");
294 
295             assertEquals(levels.length, appender.list.size(), "Expected lengths do not match number of log messages");
296 
297             for (int i = 0; i < appender.list.size(); i++) {
298                 final ILoggingEvent item = appender.list.get(i);
299 
300                 assertEquals(levels[i], item.getLevel(), "Levels not equal for element " + i);
301                 assertEquals(messages[i], item.getFormattedMessage(), "Messages not equal for element " + i);
302                 assertEquals(throwables[i], item.getThrowableProxy() != null, "Throwables not equal for elmeent " + i);
303             }
304         }
305 
306         @Override
307         public void close() throws IOException {
308             logger.detachAndStopAllAppenders();
309         }
