48      @Test
49      void testEmptyData() {
50          final byte[] empty = {};
51          final IBaseDataObject d1 = DataObjectFactory.getInstance();
52          d1.setData(empty);
53          assertTrue(DataUtil.isEmpty(d1), "empty: " + d1);
54          assertFalse(DataUtil.isNotEmpty(d1), "not empty: " + d1);
55  
56          final byte[] whitespace = {' '};
57          final byte[] whitespaces = {' ', ' '};
58          final byte[] control = {' ', ByteUtil.Ascii_DEL, ' '};
59          final byte[] foo = {'f', 'o', '1'};
60          final byte[] whiten = {' ', ' ', '1'};
61          final byte[] whitencontrol = {' ', ' ', '1', ByteUtil.Ascii_ESC};
62          final byte[] W = "Президент Буш".getBytes();
63          for (final byte[] bytes : Arrays.asList(whitespace, whitespaces, control, foo, whiten, whitencontrol, W)) {
64              assertFalse(DataUtil.isEmpty(bytes), "empty: " + Arrays.toString(bytes));
65              final IBaseDataObject d2 = DataObjectFactory.getInstance();
66              d2.setData(bytes);
67              assertFalse(DataUtil.isEmpty(d2), "empty: " + d2);
68              assertTrue(DataUtil.isNotEmpty(d2), "empty: " + d2);
69          }
70      }
