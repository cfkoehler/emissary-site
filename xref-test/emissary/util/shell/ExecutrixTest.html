93      @Test
94      void testReadWrite() throws Exception {
95          final String TMPDIR = this.e.getTmpDir();
96          assertTrue(Executrix.writeDataToFile("aaa".getBytes(), 0, 3, TMPDIR + "/foo.dat", false), "File written");
97          byte[] data = Executrix.readFile(TMPDIR + "/foo.dat");
98          assertNotNull(data, "Data must be read");
99          assertEquals(3, data.length, "Data must all be read");
100 
101         // append
102         assertTrue(Executrix.writeDataToFile("aaa".getBytes(), 0, 3, TMPDIR + "/foo.dat", true), "File written");
103         data = Executrix.readFile(TMPDIR + "/foo.dat");
104         assertNotNull(data, "Data must be read");
105         assertEquals(6, data.length, "Data must all be read");
106 
107         data = Executrix.readFile(TMPDIR + "/foo.dat", 3);
108         assertNotNull(data, "Data must be read");
109         assertEquals(3, data.length, "Data must be read up to limit");
110 
111         data = Executrix.readFile(TMPDIR + "/foo.dat", 3000);
112         assertNotNull(data, "Data must be read");
113         assertEquals(6, data.length, "Data must be read up to actual size");
114 
115         assertFalse(Executrix.writeDataToFile(null, TMPDIR + "/foo.dat"), "Should not write null data");
116         assertFalse(Executrix.writeDataToFile(null, TMPDIR + "/foo.dat", false), "Should not write null data");
117         assertFalse(Executrix.writeDataToFile(null, 0, 3, TMPDIR + "/foo.dat", false), "Should not write null data");
118 
119         assertTrue(Executrix.writeDataToFile("aaa".getBytes(), TMPDIR + "/foo.dat", false), "Overwrite longer file should truncate previous data");
120         data = Executrix.readFile(TMPDIR + "/foo.dat");
121         assertNotNull(data, "Data must be read");
122         assertEquals(3, data.length, "Data must be read up to actual size");
123 
124         data = Executrix.readDataFromFile(TMPDIR + "/foo.dat");
125         assertNotNull(data, "Data must be read");
126         assertEquals(3, data.length, "Data must be read up to actual size");
127 
128         data = Executrix.readDataFromFile(TMPDIR + "/filedoesnotexist.dat");
129         assertNull(data, "Read non existent does not throw");
130 
131         assertFalse(Executrix.writeDataToFile("aaa".getBytes(), null), "Write to null path");
132         assertFalse(Executrix.writeDataToFile("aaa".getBytes(), null, false), "Write to null path");
133         assertFalse(Executrix.writeDataToFile("aaa".getBytes(), 0, 3, null, false), "Write to null path");
134 
135         RandomAccessFile raf = new RandomAccessFile(TMPDIR + "/foo.dat", "rw");
136         data = Executrix.readDataFromFile(raf);
137         assertNotNull(data, "Data must be read from raf");
138         assertEquals(3, data.length, "Data must all be read from raf");
139         raf.close();
140 
141         raf = new RandomAccessFile(TMPDIR + "/foo.dat", "rw");
142         data = Executrix.readDataFromFile(raf, 2, 1);
143         assertNotNull(data, "Data must be read from raf");
144         assertEquals(1, data.length, "Data must all be read from raf");
145 
146         raf.seek(0);
147         data = Executrix.readDataFromFile(raf, 100, 200);
148         assertNull(data, "Data requested out of bounds");
149 
150         raf.seek(0);
151         data = Executrix.readDataFromFile(raf, 0, 100);
152         assertNotNull(data, "Available data read from raf");
153         assertEquals(3, data.length, "All available data read from raf");
154 
155         raf.seek(0);
156         final FileChannel channel = raf.getChannel();
157         data = Executrix.readDataFromChannel(channel);
158         assertNotNull(data, "Available data read from channel");
159         assertEquals(3, data.length, "All available data read from channel");
160 
161         data = Executrix.readDataFromChannel(channel, 0, 3);
162         assertNotNull(data, "Available data read from channel");
163         assertEquals(3, data.length, "All available data read from channel");
164 
165         data = Executrix.readDataFromChannel(channel, 0, 1);
166         assertNotNull(data, "Available data read from channel");
167         assertEquals(1, data.length, "All limit data read from channel");
168 
169         data = Executrix.readDataFromChannel(channel, 0, 100);
170         assertNotNull(data, "Available data read from channel");
171         assertEquals(3, data.length, "All limit data read from channel");
172 
173         data = Executrix.readDataFromChannel(channel, 50, 100);
174         assertNull(data, "Out of bounds data on channel returned non-null");
175 
176         channel.close();
177         raf.close();
178 
179         final File f = new File(TMPDIR + "/foo.dat");
180         Files.deleteIfExists(f.toPath());
181     }

257     @Test
258     void testExecute() throws IOException {
259 
260         final String[] names = this.e.makeTempFilenames();
261         logger.debug("Names for testExecute is " + Arrays.asList(names));
262 
263         final File tdir = new File(names[Executrix.DIR]);
264         Files.createDirectories(tdir.toPath());
265         assertTrue(tdir.exists() && tdir.isDirectory(), "Temp dir exists");
266 
267         assertTrue(Executrix.writeDataToFile("aaa".getBytes(), names[Executrix.INPATH]), "File written");
268         final byte[] data = Executrix.readDataFromFile(names[Executrix.INPATH]);
269         assertNotNull(data, "Data must be read from " + names[Executrix.INPATH]);
270 
271         final String cyg = System.getProperty("CYGHOME");
272         final boolean cyghome = cyg != null && cyg.contains(":");
273         final String cmd = (this.isWindows ? (cyghome ? "/bin/cp" : "copy") : "cp") + " <INPUT_NAME> <OUTPUT_NAME>";
274         String[] c = this.e.getCommand(cmd, names);
275         assertNotNull(c, "Command returned");
276         assertEquals((this.isWindows ? "cmd" : "/bin/sh"), c[0], "Command runner");
277 
278         this.e.setCommand(cmd);
279         c = this.e.getCommand(names);
280         assertNotNull(c, "Command returned");
281         assertEquals((this.isWindows ? "cmd" : "/bin/sh"), c[0], "Command runner");
282 
283         logger.debug("Command to exec is " + Arrays.asList(c));
284 
285         int pstat;
286         final StringBuffer out = new StringBuffer();
287         final StringBuffer err = new StringBuffer();
288 
289         pstat = this.e.execute(c, out, err);
290         logger.debug("Stdout: " + out);
291         logger.debug("Stderr: " + err);
292         assertTrue(pstat >= 0, "Process return value");
293         readAndNuke(names[Executrix.OUTPATH]);
294 
295         pstat = this.e.execute(c, out);
296         assertTrue(pstat >= 0, "Process return value");
297         readAndNuke(names[Executrix.OUTPATH]);
298 
299         pstat = this.e.execute(c, out, "UTF-8");
300         assertTrue(pstat >= 0, "Process return value");
301         readAndNuke(names[Executrix.OUTPATH]);
302 
303         pstat = this.e.execute(c, out, err, "UTF-8");
304         assertTrue(pstat >= 0, "Process return value");
305         readAndNuke(names[Executrix.OUTPATH]);
306 
307         final StringBuilder sout = new StringBuilder();
308         final StringBuilder serr = new StringBuilder();
309 
310         pstat = this.e.execute(c, sout);
311         assertTrue(pstat >= 0, "Process return value");
312         readAndNuke(names[Executrix.OUTPATH]);
313 
314         pstat = this.e.execute(c, sout, serr);
315         assertTrue(pstat >= 0, "Process return value");
316         readAndNuke(names[Executrix.OUTPATH]);
317 
318         pstat = this.e.execute(c, sout, serr, "UTF-8");
319         assertTrue(pstat >= 0, "Process return value");
320         readAndNuke(names[Executrix.OUTPATH]);
321 
322         final Map<String, String> env = new HashMap<>();
323         env.put("FOO", "BAR");
324 
325         pstat = this.e.execute(c, sout, serr, "UTF-8", env);
326         assertTrue(pstat >= 0, "Process return value");
327         readAndNuke(names[Executrix.OUTPATH]);
328 
329         pstat = this.e.execute(c);
330         assertTrue(pstat >= 0, "Process return value");
331         readAndNuke(names[Executrix.OUTPATH]);
332 
333         this.e.setProcessMaxMillis(0); // wait forever
334         pstat = this.e.execute(c, sout, serr, "UTF-8", env);
335         assertTrue(pstat >= 0, "Process return value");
336         readAndNuke(names[Executrix.OUTPATH]);
337 
338         assertTrue(Executrix.cleanupDirectory(tdir), "Temp area clean up removes all");
339         assertFalse(tdir.exists(), "Temp area should be gone");
340     }

342     @Test
343     void testExecuteStream() {
344 
345         String expected = "bbb";
346         final byte[] data = expected.getBytes();
347 
348         final String cmd = "/bin/cat";
349 
350         int pstat;
351         final StringBuilder sout = new StringBuilder();
352         final StringBuilder serr = new StringBuilder();
353 
354         pstat = this.e.execute(cmd, data);
355         assertTrue(pstat >= 0, "Process return value");
356 
357         pstat = this.e.execute(cmd, "".getBytes(StandardCharsets.UTF_8), sout);
358         assertTrue(pstat >= 0, "Process return value");
359         assertEquals("", sout.toString());
360         sout.setLength(0);
361 
362         pstat = this.e.execute(cmd, data, sout);
363         assertTrue(pstat >= 0, "Process return value");
364         assertEquals(expected, sout.toString().trim());
365         sout.setLength(0);
366 
367         pstat = this.e.execute(cmd, data, sout, serr);
368         assertTrue(pstat >= 0, "Process return value");
369         assertTrue(sout.toString().startsWith("bbb"));
370         assertEquals(expected, sout.toString().trim());
371         sout.setLength(0);
372 
373         pstat = this.e.execute(cmd, data, sout, serr, "UTF-8");
374         assertTrue(pstat >= 0, "Process return value");
375         assertTrue(sout.toString().startsWith("bbb"));
376         assertEquals(expected, sout.toString().trim());
377         sout.setLength(0);
378 
379         final Map<String, String> env = new HashMap<>();
380         env.put("FOO", "BAR");
381 
382         pstat = this.e.execute(new String[] {cmd}, data, sout, serr, "UTF-8", env);
383         assertTrue(pstat >= 0, "Process return value");
384         assertEquals(expected, sout.toString().trim());
385         assertEquals("", serr.toString());
386         sout.setLength(0);
387 
388         this.e.setProcessMaxMillis(0); // wait forever
389         pstat = this.e.execute(new String[] {cmd}, data, sout, serr, "UTF-8", env);
390         assertTrue(pstat >= 0, "Process return value");
391         assertEquals("", serr.toString());
392         assertEquals(expected, sout.toString().trim());
393         sout.setLength(0);
394     }
