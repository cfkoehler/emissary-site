19  class DataIdentifierTest extends UnitTest {
20  
21      byte[] DATA = new byte[1000];
22  
23      @Override
24      @BeforeEach
25      public void setUp() throws Exception {
26          super.setUp();
27          Arrays.fill(DATA, (byte) 'a');
28      }
29  
30      @Test
31      void testId() throws Exception {
32          DataIdentifier id = new DataIdentifier();
33          assertEquals(UNKNOWN_TYPE, id.identify(DATA), "Unknown default id");
34          assertTrue(id.getTestStringMaxSize() > 0, "Test string size");
35  
36          assertEquals(UNKNOWN_TYPE, id.identify(DATA), "Unknown raf id");
37          super.tearDown();
38      }
39  
40      static Stream<Arguments> arguments() {
41          return Stream.of(
42                  Arguments.of("aaa", UNKNOWN_TYPE, "Unknown id on data shorter than pattern"),
43                  Arguments.of("aaa===aaa\n\nbbb===bb", "FOO", "Valid id on data longer than pattern"),
44                  Arguments.of("ccc===ccc\n\nbbb===bbb", UNKNOWN_TYPE, "Valid id on data longer than pattern"),
45                  Arguments.of("", UNKNOWN_TYPE, "Unknown id on empty byte array"));
46      }
47  
48      @ParameterizedTest
49      @MethodSource("arguments")
50      void testIdentify(String identify, String expected, String msg) {
51          ServiceConfigGuide config = new ServiceConfigGuide();
52          config.addEntry("TYPE_FOO", "aaa===aaa");
53          DataIdentifier id = new DataIdentifier(config);
54          String result = id.identify(identify.getBytes());
55          assertEquals(expected, result, msg);
56      }
57  
58      @Test
59      void testExtensibility() {
60          DataIdentifierT id = new DataIdentifierT();
61          assertEquals(id.getTestStringMaxSize(), id.checkSize(), "String size to test");
62  
63          // Check byte array
64          assertEquals(id.checkSize(), id.checkString(DATA).length(), "Test string generation");
65          assertEquals(5, id.checkString(DATA, 5).length(), "Test string generation with length");
66          assertEquals(DATA.length, id.checkString(DATA, DATA.length + 100).length(), "Test string generation with oversize length");
67      }
68  
69  
70      private static final class DataIdentifierT extends DataIdentifier {
71          public int checkSize() {
72              return DATA_ID_STR_SZ;
73          }
74  
75          public String checkString(byte[] data) {
76              return super.getTestString(data);
77          }
78  
79          public String checkString(byte[] data, int limit) {
80              return super.getTestString(data, limit);
81          }
82      }
