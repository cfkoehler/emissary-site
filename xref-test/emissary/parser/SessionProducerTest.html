23      @Test
24      void testZoneAssignments() {
25          SimpleParser parser = new SimpleParser("This is a test".getBytes());
26          SessionProducer sp = new SessionProducer(parser, "UNKNOWN");
27          DecomposedSession d = new DecomposedSession();
28          d.setHeader("The Header".getBytes());
29          d.setFooter("The Footer".getBytes());
30          d.setData("The Data".getBytes());
31          d.setClassification("Purple Nurple");
32          d.addMetaData("foo", "bar");
33          IBaseDataObject payload = sp.createAndLoadDataObject(d, "name");
34          assertEquals("The Header", new String(payload.header()), "Incorrect header");
35          assertEquals("The Footer", new String(payload.footer()), "Incorrect footer");
36          assertEquals("The Data", new String(payload.data()), "Incorrect data");
37          assertEquals("Purple Nurple", payload.getClassification(), "Incorrect classification");
38          assertEquals("bar", payload.getStringParameter("foo"), "Incorrect metadata");
39          assertEquals("name", payload.shortName(), "Incorrect name");
40      }

42      @Test
43      void testAlternateViewAssignment() {
44          SimpleParser parser = new SimpleParser("This is a test".getBytes());
45          SessionProducer sp = new SessionProducer(parser, "UNKNOWN");
46          DecomposedSession d = new DecomposedSession();
47          d.addMetaData("FOO", "BAR");
48          d.addMetaData("ALT_VIEW_FOOVIEW", "This is the view data".getBytes());
49          d.addMetaData("ALT_VIEW_BARVIEW", "This is string data for a view");
50          IBaseDataObject payload = sp.createAndLoadDataObject(d, "name");
51          assertNotNull(payload.getAlternateView("FOOVIEW"), "Alt view must be created from byte array");
52          assertNotNull(payload.getAlternateView("BARVIEW"), "Alt veiw must be created from string");
53          assertEquals("This is the view data", new String(payload.getAlternateView("FOOVIEW")), "Byte array view data incorrect");
54          assertEquals("This is string data for a view", new String(payload.getAlternateView("BARVIEW")), "String view data incorrect");
55          assertNull(payload.getParameter("ALT_VIEW_FOOVIEW"), "View data must be removed from normal metadata");
56          assertNull(payload.getParameter("ALT_VIEW_BARVIEW"), "View data must be removed from normal metadata");
57      }
