325     @Test
326     void testRollBadFiles() throws Exception {
327         // setup
328         Path target = Files.createTempFile(targetBUDPath, "badfile", "");
329         String key = target.toString();
330         try (JournalWriter jw = new JournalWriter(targetBUDPath, key);
331                 SeekableByteChannel c = Files.newByteChannel(target, StandardOpenOption.WRITE)) {
332             ArrayList<String> strings = new ArrayList<>(BUD1_LINES);
333             strings.addAll(BUD2_LINES);
334             for (String line : strings) {
335                 c.write(ByteBuffer.wrap(line.getBytes()));
336                 jw.write(new JournalEntry(key, c.position()));
337             }
338             // phony write
339             jw.write(new JournalEntry(key, c.position() + 1000));
340 
341         }
342         long partSize = Files.size(target);
343         // open journal
344         Journal j;
345         try (JournalReader jr = new JournalReader(Paths.get(key + Journal.EXT))) {
346             j = jr.getJournal();
347         }
348         Path rolled = Files.createTempFile(targetBUDPath, "rolled_badfile", "");
349         try (SeekableByteChannel sbc = Files.newByteChannel(rolled, StandardOpenOption.WRITE)) {
350             journaledCoalescer.combineFiles(j, sbc);
351         }
352 
353         long rolledSize = Files.size(rolled);
354 
355         // verify
356         assertEquals(partSize, rolledSize);
357         assertTrue(Objects.requireNonNull(j.getLastEntry()).getOffset() > partSize);
358     }
