435         // populate the tld with all the event date options (note: year of 2016)
436         IBaseDataObject tld = new BaseDataObject();
437         tld.setParameter(EVENT_DATE, "2016-02-13 23:13:03");
438         tld.setParameter(FILE_DATE, "2016-01-05 17:45:53");
439 
440         // populate the child with all the event date options (note: year of 2015)
441         IBaseDataObject d = new BaseDataObject();
442         d.setParameter(EVENT_DATE, "2015-02-13 23:13:03");
443         d.setParameter(FILE_DATE, "2015-01-05 17:45:53");
444 
445         Date start = new Date(); // use this for relative comparison
446 
447         // hit on child EventDate
448         assertEquals("2015-02-13 23:13:03", TimeUtil.getDateAsISO8601(this.util.getEventDate(d, tld).toInstant()));
449 
450         // removing child EventDate should hit on child FILE_DATE
451         d.deleteParameter(EVENT_DATE);
452         assertEquals("2015-01-05 17:45:53", TimeUtil.getDateAsISO8601(this.util.getEventDate(d, tld).toInstant()));
453 
454         // removing child FILE_DATE should hit on tld EventDate
455         d.deleteParameter(FILE_DATE);
456         assertEquals("2016-02-13 23:13:03", TimeUtil.getDateAsISO8601(this.util.getEventDate(d, tld).toInstant()));
457 
458         // removing tld EventDate should hit on tld FILE_DATE
459         tld.deleteParameter(EVENT_DATE);
460         assertEquals("2016-01-05 17:45:53", TimeUtil.getDateAsISO8601(this.util.getEventDate(d, tld).toInstant()));
461 
462         // removing tld FILE_DATE should default to now as configured by default
463         tld.deleteParameter(FILE_DATE);
464         assertNotNull(this.util.getEventDate(d, tld));
465         assertNotEquals(-1, this.util.getEventDate(d, tld).compareTo(start));
466 
467         // changing the configuration to not default to now should return null
468         Configurator cfg = new ServiceConfigGuide();
469         final List<String> dates = new ArrayList<>();
470         dates.add(EVENT_DATE);
471         dates.add(FILE_DATE);
472         cfg.addEntries("DATE_PARAMETER", dates);
473         cfg.addEntry("DEFAULT_EVENT_DATE_TO_NOW", "false");
474         this.util = new DropOffUtil(cfg);
475 
476         assertNull(this.util.getEventDate(d, tld));
477     }
