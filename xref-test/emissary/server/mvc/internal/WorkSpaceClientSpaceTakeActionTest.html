101     @Test
102     void successfulTake() throws Exception {
103         // setup
104         WorkSpace spy = spy(new WorkSpace());
105         WorkBundle wb = new WorkBundle();
106         wb.setBundleId("1");
107         wb.addFileName("file");
108         doReturn(wb).when(spy).take(PLACE_NAME);
109         Namespace.bind(WORKSPACE_BIND_KEY, spy);
110 
111         // test
112         try (Response response = target(CLIENT_SPACE_TAKE_ACTION).request().post(Entity.form(formParams))) {
113             final int status = response.getStatus();
114             assertEquals(200, status);
115             final String result = response.readEntity(String.class);
116             final WorkBundle resultWb = WorkBundle.buildWorkBundle(result);
117             assertNotNull(resultWb);
118             assertEquals(resultWb.getFileNameList(), wb.getFileNameList());
119             assertEquals(resultWb.getBundleId(), wb.getBundleId());
