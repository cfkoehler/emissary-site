20  class StartupTest extends UnitTest {
21  
22      @Test
23      void testSortPlaces() throws IOException {
24          List<String> somePlaces = new ArrayList<>();
25          EmissaryNode node = new EmissaryNode();
26          String location = "http://" + node.getNodeName() + ":" + node.getNodePort();
27          somePlaces.add(location + "/" + CoordinationPlace.class.getSimpleName());
28          somePlaces.add(location + "/" + DelayPlace.class.getSimpleName());
29          somePlaces.add(location + "/" + DevNullPlace.class.getSimpleName());
30          somePlaces.add(location + "/" + FilePickUpPlace.class.getSimpleName());
31          somePlaces.add(location + "/" + FilePickUpClient.class.getSimpleName());
32  
33  
34          Startup startup = new Startup(node.getNodeConfigurator(), node);
35          startup.sortPlaces(somePlaces);
36          // TODO: figure out why the key isn't http://localhost:8001. Seems
37          // this way when running too, so get("") works. This test is just testing the
38          // PickUp stuff is pulled out separately
39          List<String> pickups = startup.pickupLists.get("");
40          List<String> places = startup.placeLists.get("");
41          assertIterableEquals(Arrays.asList(location + "/" + "FilePickUpPlace", location + "/" + "FilePickUpClient"), pickups);
42          assertIterableEquals(Arrays.asList(location + "/" + "CoordinationPlace", location + "/" + "DelayPlace", location + "/" + "DevNullPlace"),
43                  places);
44      }
