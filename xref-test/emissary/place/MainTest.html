212     @Test
213     void testHandleSplitOutput() throws IOException {
214         Path outputFolder = Files.createDirectories(testOutputFolder.resolve("testmain-split.dat"));
215         String[] args = {"-S", "-d", outputFolder.toRealPath().toString()};
216         Main m = new Main(this.className, args);
217         m.parseArguments();
218         try {
219             m.run();
220         } catch (Throwable t) {
221             fail("Main runner allowed exception to escape", t);
222         }
223         IBaseDataObject payload = DataObjectFactory.getInstance("aaa".getBytes(), "test", "UNKNOWN");
224         List<IBaseDataObject> atts = new ArrayList<>();
225         IBaseDataObject att1 = DataObjectFactory.getInstance("bbb".getBytes(), "safe_attempt", "SAFE_ATTEMPT");
226         atts.add(att1);
227         IBaseDataObject att2 = DataObjectFactory.getInstance("ccc".getBytes(), "escape_attempt", "./../../pwned/ESCAPE_ATTEMPT");
228         atts.add(att2);
229 
230         IBaseDataObject att3 = DataObjectFactory.getInstance("ccc".getBytes(), "attempt", "./../../testmain.dat_sibling/ESCAPE_ATTEMPT");
231         atts.add(att3);
232 
233         m.handleSplitOutput(payload, atts);
234 
235         // validate that output files "test.UNKNOWN" and "safe_attempt.SAFE_ATTEMPT" were created, but file
236         // "escape_attempt../../../pwned/ESCAPE_ATTEMPT" was not
237         String normalizedPath = outputFolder + "/" + payload.shortName() + "." + payload.currentForm();
238         assertTrue(Files.exists(Paths.get(normalizedPath).normalize()), "File \"" + normalizedPath + "\" should have been created");
239 
240         normalizedPath = outputFolder + "/" + att1.shortName() + "." + att1.currentForm();
241         assertTrue(Files.exists(Paths.get(normalizedPath).normalize()), "File \"" + normalizedPath + "\" should have been created");
242 
243         normalizedPath = outputFolder + "/" + att2.shortName() + "." + att2.currentForm();
244         assertFalse(Files.exists(Paths.get(normalizedPath).normalize()), "File \"" + normalizedPath + "\" should have NOT been created");
245 
246         normalizedPath = outputFolder + "/" + att3.shortName() + "." + att3.currentForm();
247         assertFalse(Files.exists(Paths.get(normalizedPath).normalize()), "File \"" + normalizedPath + "\" should have NOT been created");
248     }

250     @Test
251     void testFilePathIsWithinBaseDirectory() {
252 
253         String basePath = testOutputFolder.resolve("foo").toString();
254         assertEquals(basePath + "/somefile", filePathIsWithinBaseDirectory(basePath, basePath + "/somefile"));
255         assertEquals(basePath + "/otherfile", filePathIsWithinBaseDirectory(basePath, basePath + "//otherfile"));
256         assertEquals(basePath + "/foo/otherfile", filePathIsWithinBaseDirectory(basePath, basePath + "/./foo/otherfile"));
257         assertEquals(basePath + "/sub/otherfile", filePathIsWithinBaseDirectory(basePath, basePath + "/sub/././otherfile"));
258 
259         // Each of these should thrown an Exception
260         assertThrows(IllegalArgumentException.class, () -> filePathIsWithinBaseDirectory(basePath, "/var/log/somelog"));
261 
262         assertThrows(IllegalArgumentException.class,
263                 () -> filePathIsWithinBaseDirectory(basePath, basePath + "/../foo2/otherfile"),
264                 "Expected an IllegalArgumentException from input " + basePath + "/../foo2/otherfile");
265 
266         assertThrows(IllegalArgumentException.class,
267                 () -> filePathIsWithinBaseDirectory(basePath, basePath + "/../../somefile"),
268                 "Expected an IllegalArgumentException from input " + basePath + "/../../somefile");
269 
270         assertThrows(IllegalArgumentException.class,
271                 () -> filePathIsWithinBaseDirectory(basePath, basePath + "/path/../../../otherpath"),
272                 "Expected an IllegalArgumentException from input " + basePath + "/path/../../../otherpath");
273     }

275     // Override the hooks. The calls to super in these
276     // hooks are not normal, but help us mark the super class
277     // hooks as having been tested.
278     private static final class MainWithHooks extends Main {
279         public boolean[] printhook = {false, false};
280         public boolean[] arghook = {false, false};
281         public boolean[] processhook = {false, false};
282         public boolean[] splithook = {false, false};
283 
284         public MainWithHooks(String className, String[] args) {
285             super(className, args);
286         }
287 
288         @Override
289         protected boolean prePrintHook(IBaseDataObject payload, List<IBaseDataObject> att) {
290             super.prePrintHook(payload, att);
291             printhook[0] = true;
292             return false;
293         }
294 
295         @Override
296         protected void postPrintHook(IBaseDataObject payload, List<IBaseDataObject> att) {
297             super.postPrintHook(payload, att);
298             printhook[1] = true;
299         }
300 
301         @Override
302         protected boolean preArgumentsHook(CommandLine cmd) {
303             super.preArgumentsHook(cmd);
304             arghook[0] = true;
305             return true;
306         }
307 
308         @Override
309         protected void postArgumentsHook(CommandLine cmd) {
310             super.postArgumentsHook(cmd);
311             arghook[1] = true;
312         }
313 
314         @Override
315         protected boolean preProcessHook(IBaseDataObject payload, List<IBaseDataObject> attachments) {
316             super.preProcessHook(payload, attachments);
317             processhook[0] = true;
318             // Form checked in split hook test for file
319             if (payload.currentForm() == null) {
320                 payload.setCurrentForm(Form.UNKNOWN);
321             }
322             return true;
323         }
324 
325         @Override
326         protected void postProcessHook(IBaseDataObject payload, List<IBaseDataObject> att) {
327             super.postProcessHook(payload, att);
328             processhook[1] = true;
329         }
330 
331         @Override
332         protected boolean preSplitHook(IBaseDataObject payload, List<IBaseDataObject> att) {
333             super.preSplitHook(payload, att);
334             splithook[0] = true;
335             return true;
336         }
337 
338         @Override
339         protected void postSplitHook(IBaseDataObject payload, List<IBaseDataObject> att) {
340             super.postSplitHook(payload, att);
341             splithook[1] = true;
342         }
343     }
