57      private void verifyClone(final String methodName, final IBaseDataObject origObj, final Boolean isSame, final Boolean isEquals,
58              final Boolean switchWithFullClone) {
59          try {
60              final Method method = IBaseDataObject.class.getMethod(methodName);
61              final boolean isArrayType = method.getReturnType().getName().equals("[B");
62              final IBaseDataObject cloneFalseObj = IBaseDataObjectHelper.clone(origObj, false);
63              final IBaseDataObject cloneTrueObj = IBaseDataObjectHelper.clone(origObj, true);
64              verifyCloneAssertions(method, origObj, cloneFalseObj, isSame, isEquals);
65              if (switchWithFullClone) {
66                  if (isArrayType) {
67                      verifyCloneAssertions(method, origObj, cloneTrueObj, isSame, isEquals == null ? null : !isEquals);
68                  } else {
69                      verifyCloneAssertions(method, origObj, cloneTrueObj, isSame == null ? null : !isSame, isEquals == null ? null : !isEquals);
70                  }
71              } else {
72                  verifyCloneAssertions(method, origObj, cloneTrueObj, isSame, isEquals);
73              }
74          } catch (final NoSuchMethodException | SecurityException | IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
75              fail("Test error - couldn't invoke specified method", e);
76          }
77      }
78  
79      private void verifyCloneAssertions(final Method method, final Object obj1, final Object obj2, final Boolean isSame, final Boolean isEquals)
80              throws IllegalAccessException, IllegalArgumentException, InvocationTargetException {
81          final Object o1 = method.invoke(obj1);
82          final Object o2 = method.invoke(obj2);
83  
84          if (isSame != null) {
85              if (isSame) {
86                  assertSame(o1, o2);
87              } else {
88                  assertNotSame(o1, o2);
89              }
90          }
91  
92          if (isEquals != null) {
93              if (isEquals) {
94                  if (method.getReturnType().getName().equals("[B")) {
95                      assertArrayEquals((byte[]) o1, (byte[]) o2);
96                  } else {
97                      assertEquals(o1, o2);
98                  }
99              } else {
100                 assertNotEquals(o1, o2);
101             }
102         }
