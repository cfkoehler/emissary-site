58          b.addAlternateView("TESTVIEW3", "alternate view".getBytes());

60          b.addAlternateView("TESTVIEW2", null);

72          b.addAlternateView("TESTVIEW2", "alternate view".getBytes());

94      @Test
95      void testParameters() {
96          b.putParameter("ME", "YOU");
97          assertEquals("YOU", b.getStringParameter("ME"), "Gotten parameter");
98          Map<String, Object> map = new HashMap<>();
99          map.put("ONE", "uno");
100         map.put("TWO", "dos");
101         map.put("THREE", "tres");
102         b.putParameters(map);
103         assertEquals("uno", b.getStringParameter("ONE"), "Map put parameter gotten");
104         assertEquals("dos", b.getStringParameter("TWO"), "Map put parameter gotten");
105         assertEquals("tres", b.getStringParameter("THREE"), "Map put parameter gotten");
106         assertEquals("YOU", b.getStringParameter("ME"), "Gotten parameter");
107 
108         // Deletes
109         b.deleteParameter("THREE");
110         assertNull(b.getParameter("THREE"), "Deleted param is gone");
111 
112         // Overwrite
113         b.putParameter("ME", "THEM");
114         assertEquals("THEM", b.getStringParameter("ME"), "Gotten parameter");
115 
116         // Clear
117         b.clearParameters();
118         assertNull(b.getParameter("THREE"), "Deleted param is gone");
119         assertNull(b.getParameter("ME"), "Deleted param is gone");
120         Map<?, ?> m = b.getParameters();
121         assertNotNull(m, "Clear paramters leave empty map");
122         assertEquals(0, m.size(), "Clear parameters leaves empty map");
