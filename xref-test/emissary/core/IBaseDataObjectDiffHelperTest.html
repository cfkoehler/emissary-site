48      private void verifyDiff(final int expectedDifferences) {

52      private void verifyDiff(final int expectedDifferences, final DiffCheckConfiguration options) {
53          IBaseDataObjectDiffHelper.diff(ibdo1, ibdo1, differences, options);
54          assertEquals(0, differences.size());
55          IBaseDataObjectDiffHelper.diff(ibdo1, ibdo2, differences, options);
56          assertEquals(expectedDifferences, differences.size());
57          differences.clear();
58          IBaseDataObjectDiffHelper.diff(ibdo2, ibdo1, differences, options);
59          assertEquals(expectedDifferences, differences.size());
60          differences.clear();
61      }
62  
63      private void verifyDiffList(final int expectedDifferences, final List<IBaseDataObject> list1, final List<IBaseDataObject> list2) {
64          IBaseDataObjectDiffHelper.diff(list1, list1, "test", differences, EMPTY_OPTIONS);
65          assertEquals(0, differences.size());
66          IBaseDataObjectDiffHelper.diff(list1, list2, "test", differences, EMPTY_OPTIONS);
67          assertEquals(expectedDifferences, differences.size());
68          differences.clear();
69      }

75      @Test
76      void testDiffArguments() {
77          // Objects
78          checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(null, ibdo2, differences, EMPTY_OPTIONS));
79          checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(ibdo1, null, differences, EMPTY_OPTIONS));
80          checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(ibdo1, ibdo2, null, EMPTY_OPTIONS));
81          checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(ibdoList1, ibdoList2, null, differences, EMPTY_OPTIONS));
82          checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(ibdoList1, ibdoList2, "id", null, EMPTY_OPTIONS));
83          checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(ibdo1, ibdo2, differences, null));
84          checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(new Object(), new Object(), null, differences));
85          checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(new Object(), new Object(), "id", null));
86  
87          // Integers
88          checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(0, 0, null, differences));
89          checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(0, 0, "id", null));
90  
91          // Booleans
92          checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(false, false, null, differences));
93          checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(false, false, "id", null));
94  
95          // Maps
96          checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(null, new HashMap<>(), "id", differences));
97          checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(new HashMap<>(), null, "id", differences));
98          checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(new HashMap<>(), new HashMap<>(), null, differences));
99          checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(new HashMap<>(), new HashMap<>(), "id", null));
100     }

153     void testDiffParameters() {
154         final DiffCheckConfiguration checkKeyValueDiffParameter = DiffCheckConfiguration.configure().enableKeyValueParameterDiff().build();
155         final DiffCheckConfiguration checkDetailedDiffParameter = DiffCheckConfiguration.configure().enableDetailedParameterDiff().build();
156 
157         ibdo1.putParameter("STRING", "string");
158         ibdo1.putParameter("LIST", Arrays.asList("first", "second", "third"));
159 
160         verifyDiff(1);

162         verifyDiff(1, checkDetailedDiffParameter);
163 
164         ibdo1.clearParameters();
165         ibdo2.clearParameters();
166         ibdo1.putParameter("STRING", "string");
167         ibdo1.putParameter("Integer", Integer.valueOf(1));
168         ibdo2.putParameter("STRING", "string");
169 
170         verifyDiff(1);
171         verifyDiff(1, checkKeyValueDiffParameter);
172         verifyDiff(1, checkDetailedDiffParameter);
173 
174         ibdo1.clearParameters();
175         ibdo2.clearParameters();
176         ibdo1.putParameter("STRING", "string");
177         ibdo2.putParameter("STRING", "string");
178         ibdo2.putParameter("Integer", Integer.valueOf(1));
179 
180         verifyDiff(1);

182         verifyDiff(1, checkDetailedDiffParameter);

187         ibdo2.putParameter("STRING", "string");
