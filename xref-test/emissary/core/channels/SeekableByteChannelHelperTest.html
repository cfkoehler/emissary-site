34      @Test
35      void testImmutable() throws IOException {
36          final SeekableByteChannelFactory sbcf = SeekableByteChannelHelper.immutable(SeekableByteChannelHelper.memory(TEST_BYTES));
37          final SeekableByteChannel sbc = sbcf.create();
38          final ByteBuffer someTestBytes = ByteBuffer.wrap(TEST_BYTES);
39          assertThrows(NonWritableChannelException.class, () -> sbc.truncate(2l));
40          assertThrows(NonWritableChannelException.class, () -> sbc.write(someTestBytes));
41      }

43      @Test
44      void testMemory() throws IOException {
45          final SeekableByteChannelFactory sbcf = SeekableByteChannelHelper.memory(TEST_BYTES);
46          final SeekableByteChannel sbc = sbcf.create();
47          final ByteBuffer buff = ByteBuffer.allocate(4);
48          sbc.read(buff);
49          buff.flip();
50          byte[] bytes = new byte[4];
51          buff.get(bytes);
52          assertArrayEquals("test".getBytes(), bytes);

135

138         try (InputStream eis = new ExceptionInputStream(); InputStream is = new SequenceInputStream(new ByteArrayInputStream(bytes), eis)) {

143     @Test
144     void testGetDataWhenLargerThanMaxInt() throws IOException {
145         final BaseDataObject bdo = new BaseDataObject();
146         bdo.setFilename("filename.txt");
147 
148         final ListAppender<ILoggingEvent> appender = new ListAppender<>();
149         final Logger channelLogger = (Logger) LoggerFactory.getLogger(SeekableByteChannelHelper.class);
150 
151         final String testString = "test data"; // length 9
152         bdo.setChannelFactory(SeekableByteChannelHelper.memory(testString.getBytes()));
153 
154         try {
155             // Wrap the actual call with a way to read the logs
156             appender.start();
157             channelLogger.addAppender(appender);
158             final byte[] trimmedArray = SeekableByteChannelHelper.getByteArrayFromBdo(bdo, 5);
159             assertEquals(5, trimmedArray.length);
160             assertTrue(appender.list.stream().anyMatch(i -> i.getFormattedMessage().equals(
161                     "Returned data for [filename.txt] will be truncated by 4 bytes due to size constraints of byte arrays")));
162         } finally {
163             channelLogger.detachAppender(appender);
164             appender.stop();
165         }
