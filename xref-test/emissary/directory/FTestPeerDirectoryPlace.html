348         assertEquals(h.exp_placeRegistered, h.placeRegistered, "Obs reg count " + msg);
349         assertEquals(h.exp_placeDeregistered, h.placeDeregistered, "Obs dereg count " + msg);
350         assertEquals(h.exp_childAdded, h.childAdded, "Obs chadd count " + msg);
351         assertEquals(h.exp_childRemoved, h.childRemoved, "Obs chrm count " + msg);

353 
354         assertEquals(h.exp_peerUpdate, h.peerUpdate, "Obs peer count " + msg);
355         assertEquals(h.exp_lastPeerSetSize, h.lastPeerSetSize, "Obs peer size " + msg);
356     }

362     static class DOHitCounter extends DirectoryAdapter {
363         public int placeRegistered = 0;
364         public int placeDeregistered = 0;
365         public int childAdded = 0;
366         public int childRemoved = 0;
367         public int placeCostChanged = 0;
368         public int peerUpdate = 0;
369         public int lastPeerSetSize = 0;
370 
371         public int exp_placeRegistered = 0;
372         public int exp_placeDeregistered = 0;
373         public int exp_childAdded = 0;
374         public int exp_childRemoved = 0;
375         public int exp_placeCostChanged = 0;
376         public int exp_peerUpdate = 0;
377         public int exp_lastPeerSetSize = 0;
378 
379         public void clear() {
380             this.placeRegistered = 0;
381             this.placeDeregistered = 0;
382             this.childAdded = 0;
383             this.childRemoved = 0;
384             this.placeCostChanged = 0;
385             this.peerUpdate = 0;
386             this.lastPeerSetSize = 0;
387 
388             this.exp_placeRegistered = 0;
389             this.exp_placeDeregistered = 0;
390             this.exp_childAdded = 0;
391             this.exp_childRemoved = 0;
392             this.exp_placeCostChanged = 0;
393             this.exp_peerUpdate = 0;

397         public void expect(final int pr, final int pd, final int ca, final int cr, final int cce, final int pu, final int ps) {

399             this.exp_placeDeregistered = pd;
400             this.exp_childAdded = ca;
401             this.exp_childRemoved = cr;
402             this.exp_placeCostChanged = cce;
403             this.exp_peerUpdate = pu;
404             this.exp_lastPeerSetSize = ps;
405         }
