38  /**
39   * The Rollable implemenation that uses a journal to record offsets of completed writes to a pool of outputs. The
40   * Journal serves as a write ahead log and records positions of all open file handles until rolled.
41   * <p>
42   * During a roll, all Journals are identified and their outputs are combined into a destination filename denoted by the
43   * FileNameGenerator.
44   *
45   */
46  public class JournaledCoalescer implements IJournaler, ICoalescer {

100     /**
101      * The Rollable with take all files in a Path and combine them into a single destination file on each roll.
102      *
103      * @param outPath The Path to use for reading input and writing combined output
104      * @param fileNameGenerator The FileNameGenerator to use for unique destination file names
105      * @param poolsize The max number of outputs for the pool.
106      */
107     public JournaledCoalescer(final Path outPath, final FileNameGenerator fileNameGenerator, int poolsize) throws IOException, InterruptedException {
108         this.outputPath = outPath.toAbsolutePath();
109         this.fileNameGenerator = fileNameGenerator;
110         this.poolsize = poolsize;
111         validateOutputPath();
112         cleanupOrphanedRolledFiles();
113         initializeNextPool();
114     }
115 
116     /**
117      * Validate the Path we are using for combining files
118      */
119     private void validateOutputPath() throws IOException {
120         if (!exists(this.outputPath)) {
121             throw new FileNotFoundException("The output file path does not exist: " + this.outputPath + ".");
122         } else if (!isDirectory(this.outputPath)) {
123             throw new IllegalArgumentException("The output file path is not a directory: " + this.outputPath + ".");
124         } else if (!(isReadable(this.outputPath) && isWritable(this.outputPath))) {
125             throw new IllegalAccessError("The output path is not readable and writable: " + this.outputPath + ".");
126         }
127     }

129     /**
130      * Sometimes the rolled files can hang around after a crash. If there is a rolled file, that means all of the files
131      * coalesced successfully but cleanup failed. If there are not any files with the same name, just rename the rolled
132      * file. Otherwise, the rolled file will get cleaned up with the normal process.
133      */
134     private void cleanupOrphanedRolledFiles() throws IOException {
135         try (DirectoryStream<Path> stream = Files.newDirectoryStream(outputPath, "*" + ROLLED_EXT)) {
136             for (Path entry : stream) {
137                 String finalOutputFilename = FilenameUtils.getBaseName(entry.toString());
138                 if (isOrphanedFile(finalOutputFilename)) {
139                     finalize(entry, outputPath.resolve(finalOutputFilename));
140                 }
141             }
142         } catch (IOException e) {
143             LOG.error("There was an error trying to cleanup rolled files {}", outputPath, e);
144         }
145     }

147     /**
148      * Sets the current file name to be used when creating files and rolling output. Closes current pool, retrieves a list
149      * of Journal files that need to be rolled, and instantiates the next pool.
150      * <p>
151      * Called by the roll method, synchronized for consistency.
152      */
153     private Collection<Path> initializeNextPool() throws IOException, InterruptedException {
154         lock.lock();
155         try {
156             if (journaledPool != null) {
157                 this.journaledPool.close();
158             }
159             Collection<Path> journals = JournalReader.getJournalPaths(outputPath);
160             this.journaledPool = new JournaledChannelPool(outputPath, this.fileNameGenerator.nextFileName(), poolsize);
161             LOG.debug("Generated new Journal file name: {}", this.journaledPool);
162             return journals;
163         } finally {
164             lock.unlock();
165         }
166     }

168     /**
169      * Returns and KeyedOutput object containing the final output file and can be written to as either an OutputStream or a
170      * SeekableByteChannel. This method will block if objects from the pool have been exhausted.
171      *
172      * @return a KeyedOutput
173      */
174     @Override
175     public final KeyedOutput getOutput() throws IOException {
176         lock.lock();
177         try {
178             // Return the final, full path
179             return journaledPool.getFree();
180         } catch (InterruptedException ex) {
181             Thread.currentThread().interrupt();
182             throw new IOException("Interrupted trying to obtain KeyedOutput", ex);
183         } finally {
184             lock.unlock();
185         }
186     }

188     @Override
189     public void close() throws IOException {
190         roll();

193     @Override
194     public void roll() {
195         try {
196             this.rolling = true;
197             Collection<Path> paths = initializeNextPool();
198             coalesce(paths);
199         } catch (IOException ex) {
200             LOG.error("Error occurred during roll.", ex);
201         } catch (InterruptedException ex) {
202             LOG.warn("Roll interrupted during execution. Should continue on next roll.", ex);
203             Thread.currentThread().interrupt();
204         } finally {
205             this.rolling = false;
206         }
207     }

214     @Override
215     public void coalesce(Collection<Path> journalPaths) throws IOException {
216         if (journalPaths.isEmpty()) {
217             // nothing to do...
218             return;
219         }
220         HashMap<String, Collection<Journal>> outputMap = new HashMap<>();
221         journalPaths.forEach(path -> loadJournal(path, outputMap));
222         outputMap.forEach(this::coalesceFiles);
223     }

225     private void loadJournal(Path path, HashMap<String, Collection<Journal>> outputMap) {
226         try (JournalReaderrnalReader.html#JournalReader">JournalReader jr = new JournalReader(path)) {
227             Journal j = jr.getJournal();
228             outputMap.computeIfAbsent(j.getKey(), k -> new ArrayList<>()).add(j);
229         } catch (IOException ex) {
230             LOG.error("Unable to load Journal {}, renaming to {}{}", path.toString(), path.toString(), ERROR_EXT, ex);
231             renameToError(path);
232         }
233     }

235     /**
236      * Combines all files in the path that are ready to be rolled into their output file.
237      *
238      * @param key The path to use for reading and writing files
239      * @param journals The journal files currently needed to roll
240      */
241     protected void coalesceFiles(String key, Collection<Journal> journals) {
242         try {
243             // Create the path to the final outputFile
244             Path finalOutputPath = this.outputPath.resolve(key);
245             Path rolledOutputPath = this.outputPath.resolve(key + ROLLED_EXT);
246 
247             // Check to see if we already rolled files successfully and crashed on deletion
248             if (Files.exists(rolledOutputPath)) {
249                 LOG.warn("Full output file already found {}. Deleting old part files.", rolledOutputPath);
250                 finalize(journals, rolledOutputPath, finalOutputPath);
251                 return;
252             }
253 
254             // Create the path to the working outputFile
255             Path workingOutputPath = this.outputPath.resolve(key + ROLLING_EXT);
256 
257             // Create the working file output stream, truncating a bad file from a crashed run, if it exists
258             try (FileChannel workingOutputChannel = FileChannel.open(workingOutputPath, CREATE, TRUNCATE_EXISTING, WRITE)) {
259                 // Combine the files into the rolledOutputFile and delete them
260                 for (Journal j : journals) {
261                     combineFiles(j, workingOutputChannel);
262                 }
263                 // Flush and close output stream
264                 workingOutputChannel.force(true);
265             }
266 
267             Files.move(workingOutputPath, rolledOutputPath);
268             LOG.info("Successfully coalesced {} files into: {}. Size: {}", journals.size(), rolledOutputPath, Files.size(rolledOutputPath));
269 
270             finalize(journals, rolledOutputPath, finalOutputPath);
271         } catch (IOException ex) {
272             LOG.error("IOException while processing journals for {}", key, ex);
273         }
274     }

276     /**
277      * Copies all bytes from all paths that match to an output stream.
278      *
279      * @param journal The journal to combine in the output stream
280      * @param rolledOutput The OutputStream object to use
281      */
282     protected void combineFiles(Journal journal, SeekableByteChannel rolledOutput) throws IOException {
283         long startPos = rolledOutput.position();
284         JournalEntry last = journal.getLastEntry();
285         if (last == null) {
286             LOG.debug("Empty Journal encountered. {}", journal);
287             return;
288         }
289         long offset = last.getOffset();
290         Path p = Paths.get(last.getVal());
291         LOG.debug("Reading from path {}", p);
292         try (FileChannel part = FileChannel.open(p, READ)) {
293             long partSize = Files.size(p);
294             if (partSize < last.getOffset()) {
295                 JournalEntry lastGood = journal.getLastValidEntry(partSize);
296                 offset = lastGood.getOffset();
297                 LOG.warn("The bgpart file, {}, likely lost data due to a crash. Part size: {}, Expected {}, Actual: {}", last.getVal(), partSize,
298                         last.getOffset(), offset);
299             }
300             long xfer;
301             // for loop due to contract of channel.transferTo()
302             for (long count = offset; count > 0L;) {
303                 xfer = part.transferTo(part.position(), count, rolledOutput);
304                 part.position(part.position() + xfer);
305                 count -= xfer;
306                 if (part.position() == partSize && count > 0L) {
307                     throw new IOException("Premature EOF. Expected " + offset + ", but only transferred " + partSize);
308                 }
309             }
310             LOG.debug("Successfully appended {} bytes from {} to output file.", offset, p);
311         } catch (IOException ex) {
312             LOG.error("Exception attempting to transfer {} bytes from {} to output", offset, p.toString(), ex);
313             renameToError(p);
314             renameToError(journal.getJournalPath());
315             rolledOutput.truncate(startPos);
316             rolledOutput.position(startPos);
317         }
318     }

325     protected void finalize(Collection<Journal> journals, Path rolledOutputPath, Path finalOutputPath) throws IOException {
326         cleanupFiles(journals);
327         finalize(rolledOutputPath, finalOutputPath);

330     protected void finalize(Path rolledOutputPath, Path finalOutputPath) throws IOException {
331         if (!Files.exists(rolledOutputPath)) {
332             return;
333         }
334 
335         if (Files.size(rolledOutputPath) > 0) {
336             Files.move(rolledOutputPath, finalOutputPath);
337             LOG.info("Cleaned part files and moved rolled file to {}", finalOutputPath);
338         } else {
339             // delete the rolled file if it is empty
340             Files.delete(rolledOutputPath);
341         }
342     }
343 
344     protected void renameToError(Path path) {
345         try {
346             Path errorPath = Paths.get(path.toString() + ERROR_EXT);
347             Files.move(path, errorPath);
348         } catch (IOException ex) {
349             LOG.warn("Unable to rename file {}.", path.toString(), ex);
350         }
351     }

353     /**
354      * Test to see if a rolled file is orphaned
355      *
356      * @param startsWith the name of the rolled file
357      * @return true if no part/journal files exist, false otherwise
358      * @throws IOException if there is an issue
359      */
360     private boolean isOrphanedFile(String startsWith) throws IOException {
361         try (DirectoryStream<Path> stream = Files.newDirectoryStream(outputPath, startsWith + PART_GLOB)) {
362             return !stream.iterator().hasNext();
363         }
364     }

366     protected void cleanupFiles(Collection<Journal> journals) throws IOException {
367         for (Journal journal : journals) {
368             Path jpath = journal.getJournalPath();
369             deleteParts(journal.getEntries());
370             Files.deleteIfExists(jpath);
371         }
372     }
373 
374     private static void deleteParts(Collection<JournalEntry> entries) throws IOException {
375         for (JournalEntry entry : entries) {
376             Path p = Paths.get(entry.getVal());
377             Files.deleteIfExists(p);
378         }
