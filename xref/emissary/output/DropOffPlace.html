93      protected void configurePlace() {
94          // Set configuration info on file paths
95          this.dropOffUtil = new DropOffUtil(configG);
96          this.doSynchronized = configG.findBooleanEntry("SYNCHRONIZED_PROCESS", false);
97          this.failurePolicyTerminate = configG.findBooleanEntry("FAILURE_TERMINATES_CHAIN", true);
98          this.outputObjectMetrics = configG.findBooleanEntry("OUTPUT_OBJECT_METRICS", false);
99          this.objectMetricsFields = configG.findEntries("OBJECT_METRICS_FIELDS");
100         this.outputObjectMetricsLatency = configG.findStringEntry("OBJECT_METRICS_LATENCY", "");
101         String outputTldMetricsLatencyFormat = configG.findStringEntry("OBJECT_METRICS_LATENCY_FORMAT", "yyyy-MM-dd'T'HH:mm:ss.SSSXXX");
102         this.latencyDateTimeFormatter = DateTimeFormatter.ofPattern(outputTldMetricsLatencyFormat);
103 
104         // Build and store all the filter that are desired IN THE ORDER SPECIFIED
105         final List<String> filterClasses = configG.findEntries("OUTPUT_FILTER");
106         initializeFilters(filterClasses);
107 
108         objectMetricsLog = LoggerFactory.getLogger("objectMetrics");
109     }

111     /**
112      * Start up the requested filter
113      * 
114      * @param filterClasses the name:class values of the configured filter for this drop off
115      */
116     protected void initializeFilters(final List<String> filterClasses) {
117         for (final String entry : filterClasses) {
118             final String name;
119             final String clazz;
120             Configurator filterConfig = null;
121             final int colpos = entry.indexOf(':');
122             if (colpos > -1) {
123                 name = entry.substring(0, colpos);
124                 clazz = entry.substring(colpos + 1);
125                 final String filterConfigName = configG.findStringEntry(name);
126                 if (filterConfigName != null) {
127                     try {
128                         filterConfig = ConfigUtil.getConfigInfo(filterConfigName);
129                     } catch (IOException configError) {
130                         logger.warn("Specified filter configuration {} cannot be loaded", filterConfigName);
131                         continue;
132                     }
133                 }
134             } else {
135                 name = null;
136                 clazz = entry;
137             }
138 
139             try {
140                 final Object filter = emissary.core.Factory.create(clazz);
141                 if (filter != null && filter instanceof IDropOffFilter) {
142                     final IDropOffFilter../emissary/output/filter/IDropOffFilter.html#IDropOffFilter">IDropOffFilter f = (IDropOffFilter) filter;
143                     f.initialize(configG, name, filterConfig);
144                     addFilter(f);
145                 } else {
146                     logger.error("Misconfigured filter {} is not an IDropOffFilter instance, ignoring it", clazz);
147                 }
148             } catch (Exception ex) {
149                 logger.error("Unable to create or initialize {}", clazz, ex);
150             }
151         }
152 
153         // Collect the set of content types to elide
154         this.elideContentForms = configG.findEntriesAsSet("ELIDE_CONTENT");
155 
156         // collect the set of no-nuke forms
157         this.noNukeForms = configG.findEntriesAsSet("NO_NUKE_FORM");
158 
159         if (logger.isInfoEnabled()) {
160             logger.debug("Setting ELIDE_CONTENT forms to " + this.elideContentForms);
161             final StringBuilder sb = new StringBuilder("Output Filters:");
162             if (this.outputFilters.size() > 0) {
163                 for (final IDropOffFilter f : this.outputFilters) {
164                     sb.append(" ").append(f.getFilterName()).append("(").append(f.getClass().getName()).append(")");
165                 }
166             } else {
167                 sb.append(" NONE!");
168             }
169             logger.info(sb.toString());
170         }
171 
172         if (logger.isDebugEnabled()) {
173             final IBaseDataObject fakePayload = DataObjectFactory.getInstance(new byte[0], "fakename", Form.UNKNOWN);
174             for (final IDropOffFilter filter : getFilters()) {
175                 final String name = filter.getFilterName();
176                 final String spec = filter.getOutputSpec();
177 
178                 logger.debug("Adding filter={}, spec={}, sample={}, class={}", name, spec, this.dropOffUtil.getPathFromSpec(spec, fakePayload),
179                         filter.getClass().getSimpleName());
180             }
181         }
182     }

194     /**
195      * "HD" agent calls this method when visiting the place. If you use {@link emissary.core.MobileAgent} this method is
196      * never called. This method overrides {@link ServiceProviderPlace} and allows this processing place to have access to
197      * all payloads wanting to be dropped off in a single list.
198      * 
199      * @param payloadList list of IBaseDataObject from an {@link emissary.core.HDMobileAgent}
200      */
201     @Override
202     public List<IBaseDataObject> agentProcessHeavyDuty(final List<IBaseDataObject> payloadList) throws Exception {
203 
204         logger.debug("Entering DropOffPlace.agentProcessHeavyDuty with {} payload items", payloadList.size());
205 
206         // Prepare each incoming payload object
207         for (final IBaseDataObject d : payloadList) {
208             try {
209                 // checking to see if any object in the tree is marked as not outputable
210                 if (!d.isOutputable()) {
211                     logger.info("Skipping object since it is not able to be output ID:{}", this.dropOffUtil.getBestId(d, payloadList.get(0)));
212                     return Collections.emptyList();
213                 }
214 
215                 // Process the payload item with HDcontext=true
216                 processData(d, true);
217             } catch (Exception e) {
218                 logger.error("Place.process threw:", e);
219                 d.addProcessingError("agentProcessHD(" + myKey + "): " + e);
220 
221                 if (!d.currentForm().equals(Form.ERROR)) {
222                     d.pushCurrentForm(Form.ERROR);
223                 }
224             }
225         }
226 
227         // Prepare the data and metadata for filter output
228         final Map<String, Object> filterParams = new HashMap<>();
229         preFilterHook(payloadList, filterParams);
230 
231         // Run the filter on the output, indicating that
232         // the records are pre-sorted, if the filter cares
233         runOutputFilters(payloadList, filterParams);
234 
235         // Any cleanup operations needed
236         postFilterHook(payloadList, filterParams);
237 
238         if (!payloadList.isEmpty()) {
239             // Should have been sorted by the prefilter hook
240 
241             // Just report the TLD object ID
242             final IBaseDataObject tld = payloadList.get(0);
243 
244             if (outputObjectMetrics) {
245                 objectMetricsLog.info(appendEntries(outputObjectMetrics(tld, objectMetricsFields)), "Finished DropOff");
246             }
247 
248             logger.info("Finished DropOff for object {}, with external id: {}, with total processing time: {}ms, with filetype: {}",
249                     tld.getInternalId(), this.dropOffUtil.getBestId(tld, tld), (new Date().getTime() - tld.getCreationTimestamp().getTime()),
250                     tld.getFileType());
251         }
252 
253         // Execute 'Dispose Runnables' to tidy up resources used with SeekableByteChannelFactory implementations
254         DisposeHelper.execute(payloadList);
255 
256         // This place does not sprout, return an empty list
257         return Collections.emptyList();
258     }

260     /**
261      * Called by MobileAgent through ServiceProviderPlace to handle a single payload
262      * 
263      * @param tData the payload to work on
264      */
265     @Override
266     public void process(final IBaseDataObject tData) {
267         if (DataUtil.isEmpty(tData)) {
268             logger.warn("null/empty data object");
269             return;
270         }
271 
272         // checking to see if the object is marked as not outputable
273         if (!tData.isOutputable()) {
274             logger.warn("Skipping object since it is not able to be output ID:{}", this.dropOffUtil.getBestId(tData, tData));
275             return;
276         }
277 
278         // synchronization can be set by config file entry
279         if (this.doSynchronized) {
280             synchronized (this) {
281                 processData(tData, false);
282             }
283         } else {
284             processData(tData, false);
285         }
286 
287         // Execute 'Dispose Runnables' to tidy up resources used with SeekableByteChannelFactory implementations
288         DisposeHelper.execute(tData);
289     }

292      * Prepare a list of payload object to be filtered
293      * 
294      * @param payloadList the list of items that were eligible for output
295      * @param filterParams metadata needed for the output filter
296      */
297     public void preFilterHook(final List<IBaseDataObject> payloadList, final Map<String, Object> filterParams) {
298         // Sort the list of records
299         Collections.sort(payloadList, new ShortNameComparator());
300         filterParams.put(IDropOffFilter.PRE_SORTED, Boolean.TRUE);
301         filterParams.put(IDropOffFilter.TLD_PARAM, payloadList.get(0));
302 
303         // Prepare the metadata
304         this.dropOffUtil.processMetadata(payloadList);
305     }

307     /**
308      * Clean up after all filter are done
309      * 
310      * @param payloadList the list of items that were eligible for output
311      * @param filterParams metadata needed for the output filter
312      */
313     public void postFilterHook(final List<IBaseDataObject> payloadList, final Map<String, Object> filterParams) {
314         // remove the current forms we used or could have used
315         for (final IBaseDataObject dataObject : payloadList) {
316             // Save off no-nuke forms
317             final List<String> saveForms = new ArrayList<>();
318             for (final String nnf : this.noNukeForms) {
319                 if (dataObject.searchCurrentForm(nnf) > -1) {
320                     saveForms.add(nnf);
321                 }
322             }
323             // nuke 'em
324             this.nukeMyProxies(dataObject);
325 
326             // Restore the no-nukes
327             for (final String sf : saveForms) {
328                 dataObject.pushCurrentForm(sf);
329             }
330         }
331 
332     }

334     /**
335      * Internal method to process a single data object
336      * 
337      * @param tData the payload to work on or prepare
338      * @param haveList true if in HD context
339      */
340     protected void processData(final IBaseDataObject tData, final boolean haveList) {
341 
342         logger.debug("DropOff is working on {}, current form is {}", tData.shortName(), tData.getAllCurrentForms());
343 
344         final StringBuilder poppedForms = new StringBuilder();
345 
346         String prevBin = "";
347 
348         // skip the I/O for some types for all filter
349         for (int i = 0; i < tData.currentFormSize(); i++) {
350             final String cf = tData.currentFormAt(i);
351             if (this.elideContentForms.contains(cf)) {
352                 tData.setData(("[[ " + tData.getAllCurrentForms() + " content elided in DropOffPlace. ]]").getBytes());
353             }
354         }
355 
356         // Write out data for all the destinations we area proxy for, popping
357         // them off the stack as they are handled.
358 
359         final Set<String> serviceProxies = getProxies();
360         final Set<String> cfSet = new HashSet<>();
361         for (int i = 0; i < tData.currentFormSize(); i++) {
362             final String cf = tData.currentFormAt(i);
363 
364             if (serviceProxies.contains(cf) || serviceProxies.contains("*")) {
365                 // Record extra drop offs in the transform history since
366                 // we are taking precedence over the normal agent/directory mechanism
367                 // Do this before popping the extra destination so that it
368                 // is not just lost forever. Since the agent appends
369                 // one entry to the history when it sends the agent here,
370                 // that's the top current form and
371                 // we don't want to duplicate that one
372 
373                 if (!prevBin.equals(cf) && (i > 0) && !cfSet.contains(cf) && !("UNKNOWN".equals(cf) || cf.endsWith("-PROCESSED"))) {
374                     // e.g.: [dataType].DROP_OFF.IO.host.dom:port/DropOffPlace
375                     final DirectoryEntry de = getDirectoryEntry();
376                     de.setDataType("[" + cf + "]");
377                     tData.appendTransformHistory(de.getKey());
378                 }
379 
380                 // Accumulate forms we have handled in poppedForms
381                 if (poppedForms.length() > 0) {
382                     poppedForms.append(" ");
383                 }
384                 poppedForms.append(cf);
385                 cfSet.add(cf);
386 
387                 prevBin = cf;
388             }
389         }
390 
391         // Record the list of forms
392         tData.setParameter("POPPED_FORMS", poppedForms.toString());
393 
394         // Do the output filtering now if we aren't in HD mode
395         if (!haveList) {
396 
397             // There currently are no extra params needed
398             final Map<String, Object> filterParams = new HashMap<>();
399 
400             runOutputFilters(tData, filterParams);
401 
402             // Actually remove the current forms we used or could have used
403             this.nukeMyProxies(tData);
404 
405             logger.debug("DropOff finished with {}", tData.shortName());
406         }
407     }

409     /**
410      * Run all the output filter
411      * 
412      * @param target either IBaseDataObject or List thereof
413      * @param filterParams other parameters that filter need
414      */
415     @SuppressWarnings("unchecked")
416     protected void runOutputFilters(final Object target, final Map<String, Object> filterParams) {
417 
418         IBaseDataObject doTarget = null;
419         List<IBaseDataObject> listTarget = null;
420         if (target instanceof IBaseDataObject) {
421             doTarget = (IBaseDataObject) target;
422         } else if (target instanceof List) {
423             listTarget = (List<IBaseDataObject>) target;
424         } else {
425             logger.error("Cannot run filter on {}", target.getClass().getName());
426             return;
427         }
428 
429         // Write output onto each of the filter that have been
430         // configured, as long as they work
431         for (final IDropOffFilter filter : this.outputFilters) {
432             final long start = System.currentTimeMillis();
433 
434             // call the filter to output its data
435             int filterStatus = IDropOffFilter.STATUS_FAILURE;
436             try {
437                 if (listTarget != null && filter.isOutputtable(listTarget)) {
438                     filterStatus = filter.filter(listTarget, filterParams);
439                 } else if (doTarget != null && filter.isOutputtable(doTarget)) {
440                     filterStatus = filter.filter(doTarget, filterParams);
441                 } else {
442                     logger.debug("Filter {} not Outputtable for {}", filter.getFilterName(), listTarget != null ? "list" : "single payload");
443                     filterStatus = IDropOffFilter.STATUS_SUCCESS;
444                 }
445                 logger.debug("Filter {} took {}s - {}", filter.getFilterName(), ((System.currentTimeMillis() - start) / 1000.0), filterStatus);
446             } catch (Exception e) {
447                 logger.error("Filter {} failed", filter.getFilterName(), e);
448             }
449 
450             if ((filterStatus != IDropOffFilter.STATUS_SUCCESS) && this.failurePolicyTerminate) {
451                 logger.error("DropOff Filter chain terminated at {} due to error return status", filter.getFilterName());
452                 break;
453             }
454         }
455     }

493     /**
494      * Create a one line json output of specified parameters
495      *
496      * @param ibdo IBaseDataObject for the Top Level Document
497      * @param outputParams List of parameters to output
498      */
499     protected Map<String, String> outputObjectMetrics(IBaseDataObject ibdo, List<String> outputParams) {
500         // Create output map
501         Map<String, String> jsonMap = new HashMap<>();
502         jsonMap.put("InternalId", ibdo.getInternalId().toString());
503 
504         long processingLatency;
505         if (outputObjectMetricsLatency.isEmpty() || !ibdo.hasParameter(outputObjectMetricsLatency)) {
506             processingLatency = Instant.now().toEpochMilli() - ibdo.getCreationTimestamp().getTime();
507         } else {
508             processingLatency = Instant.now().toEpochMilli()
509                     - Instant.from(latencyDateTimeFormatter.parse(ibdo.getStringParameter(outputObjectMetricsLatency))).toEpochMilli();
510         }
511         jsonMap.put("ProcessingLatency", String.valueOf(processingLatency));
512 
513         // Add other specified Parameters
514         for (String param : outputParams) {
515             if (ibdo.hasParameter(param)) {
516                 jsonMap.put(param, ibdo.getStringParameter(param));
517             }
518         }
519         return jsonMap;
520     }
