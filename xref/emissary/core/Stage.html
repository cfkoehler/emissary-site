1   package emissary.core;

3   import java.util.ArrayList;
4   import java.util.List;
5   
6   /**
7    * Information class about processing stages in the workflow
8    */
9   public class Stage {
10      // List of stages to use. Use non-default constructors or extend class to change
11      protected static List<String> stages;
12  
13      // Each stage must be marked parallel or not
14      protected static List<Boolean> parallel;
15  
16      public Stage() {}
17  
18      static {
19          stages = new ArrayList<>();
20          parallel = new ArrayList<>();
21          stages.add("STUDY"); // prepare, coordinate idents
22          parallel.add(Boolean.FALSE);
23  
24          stages.add("ID"); // identification phase
25          parallel.add(Boolean.FALSE);
26  
27          stages.add("COORDINATE"); // Coordinate processing
28          parallel.add(Boolean.FALSE);
29  
30          stages.add("PRETRANSFORM"); // before transform hook
31          parallel.add(Boolean.TRUE);
32  
33          stages.add("TRANSFORM"); // transformation phase
34          parallel.add(Boolean.FALSE);
35  
36          stages.add("POSTTRANSFORM"); // after transform hook
37          parallel.add(Boolean.TRUE);
38  
39          stages.add("ANALYZE"); // analysis/metadata generation
40          parallel.add(Boolean.TRUE);
41  
42          stages.add("VERIFY"); // verify for output
43          parallel.add(Boolean.FALSE);
44  
45          stages.add("IO"); // output
46          parallel.add(Boolean.FALSE);
47  
48          stages.add("REVIEW"); // finish off
49          parallel.add(Boolean.FALSE);
50      }
51  
52  
53      /**
54       * Indicate if a stage supports parallel processing or not
55       * 
56       * @param i the index of the stage
57       */
58      public boolean isParallelStage(final int i) {
59          if (i >= 0 && i < parallel.size()) {
60              return parallel.get(i);
61          }
62          return false;
63      }
64  
65      /**
66       * Indicate if named stages supports parallel processing or not
67       * 
68       * @param stage the name of the stage
69       */
70      public boolean isParallelStage(final String stage) {
71          for (int i = 0; i < stages.size(); i++) {
72              if (stages.get(i).equals(stage)) {
73                  return parallel.get(i);
74              }
75          }
76          return false;
77      }
78  
79  
80      /**
81       * Return stage name at index
82       */
83      public String getStageName(final int i) {
84          if (i >= 0 && i < stages.size()) {
85              return stages.get(i);
86          } else {
87              return "UNDEFINED";
88          }
89      }
90  
91      /**
92       * Return stage index for name
93       * 
94       * @param stage the name of the stage
95       * @return the index of the stage or -1 if no such stage
96       */
97      public int getStageIndex(final String stage) {
98          for (int i = 0; i < stages.size(); i++) {
99              if (stages.get(i).equals(stage)) {
100                 return i;
101             }
102         }
103         return -1;
104     }
105 
106 
107     /**
108      * Return a list of stages
109      */
110     public List<String> getStages() {
111         return new ArrayList<>(stages);
112     }
113 
114     /**
115      * Return the stage following the named stage or null if the end of the list
116      * 
117      * @param stage the one to find following for
118      */
119     public String nextStageAfter(final String stage) {
120         for (int i = 0; i < stages.size(); i++) {
121             if (stages.get(i).equals(stage) && (i < stages.size() - 1)) {
122                 return stages.get(i + 1);
123             }
124         }
125         return null;
126     }
127 
128     /**
129      * Get size of stage list
130      */
131     public int size() {
132         return stages.size();
133     }
134 }
