100     /**
101      * Escape a string so it is suitable for use in a CSV record
102      */
103     public static String csvescape(final String field) {
104         final String SEP = ",";
105         String s = field;
106         if ((s != null) && (s.contains("\n") || s.contains("\r"))) {
107             s = NL_REPL.matcher(s).replaceAll(" ");
108         }
109 
110         if ((s != null) && (s.contains(SEP) || s.indexOf('"') > -1)) {
111             s = "\"" + s.replace("\"", "\"\"") + "\""; // escape quotes
112         }
113 
114         return s;
115     }

117     /**
118      * Get the event or collection date from a data object by checking the EventDate and FILE_DATE parameters. If both are
119      * missing default to today.
120      * 
121      * @param payload data object to examine
122      * @return event date
123      */
124     @Deprecated
125     public static Calendar getEventDate(final IBaseDataObject payload) {
126         String evDate = payload.getStringParameter(EVENT_DATE);
127         if (evDate == null) {
128             evDate = payload.getStringParameter(FILE_DATE);
129         }
130         Date eventDate = null;
131         if (evDate != null) {
132             try {
133                 eventDate = emissary.util.TimeUtil.getDateFromISO8601(evDate);
134             } catch (DateTimeParseException e) {
135                 logger.debug("Could not parse event date from payload, default to curren time", e);
136                 eventDate = new Date();
137             }
138         } else {
139             eventDate = new Date();
140         }
141         return getCal(eventDate);
142     }

201     /**
202      * Join a collection with a delimiter
203      * 
204      * @param values the collection to join
205      * @param delim the string delimiter to use
206      * @param sort true if the output should be sorted
207      */
208     @Deprecated
209     public static String join(final Collection<String> values, final String delim, final boolean sort) {
210         final Collection<String> tojoin;
211         if (sort) {
212             tojoin = new TreeSet<>();
213             tojoin.addAll(values);
214         } else {
215             tojoin = values;
216         }
217 
218         final StringBuilder sb = new StringBuilder();
219         for (final String v : tojoin) {
220             if (sb.length() > 0) {
221                 sb.append(delim);
222             }
223             sb.append(v);
224         }
225         return sb.toString();
226     }

262     /**
263      * Assume key is name of String value parameter. Check wither parent already has string value in a delim-separated
264      * string. If so do nothing, otherwise, add delim + value
265      * 
266      * @param parent IBaseDataObject to mere String parameter with
267      * @param key name of String parameter
268      * @param val String to append if not already in parents values
269      */
270     @Deprecated
271     public static void mergeParameter(final IBaseDataObject parent, final String key, final String val) {
272         String param;
273         if (parent.hasParameter(key)) {
274             param = parent.getStringParameter(key);
275             final Set<String> oldvals = split(param);
276             final Set<String> newvals = split(val);
277             newvals.addAll(oldvals);
278             param = join(newvals);
279         } else {
280             param = val;
281         }
282         parent.setParameter(key, param);
283     }
