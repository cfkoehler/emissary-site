7   package emissary.util.shell;
8   
9   import org.slf4j.MDC;
10  
11  import java.util.Map;
12  
13  /**
14   *
15   * @author ce
16   * @version 1.0
17   */
18  public abstract class ProcessReader extends Thread {
19  
20      private Map<String, String> contextMap;
21  
22      public abstract void finish();
23  
24      /**
25       * Allows parent threads to pass state values retrieved through a call to {@link MDC#getCopyOfContextMap()}
26       * 
27       * @param contextMap Map of captured point-in-time state values
28       */
29      public void setContextMap(Map<String, String> contextMap) {
30          this.contextMap = contextMap;
31      }
32  
33      /**
34       * Applies logger context to the current thread. This method should only be called from within the overridden
35       * {@link Thread#run()} method, since the context is backed by a {@link ThreadLocal ThreadLocal&lt;T&gt;} map.
36       */
37      protected final void applyLogContextMap() {
38          if (contextMap != null) {
39              MDC.setContextMap(contextMap);
40          }
41      }
42  
43      /**
44       * Wrapper method to ensure that the {@link #applyLogContextMap()} method is always invoked before executing the core
45       * functionality of ProcessReader subclasses
46       */
47      public void run() {
48          applyLogContextMap();
49          runImpl();
50      }
51  
52      /**
53       * Abstract method that subclasses should override to implement their core thread-specific functionality
54       */
55      abstract void runImpl();
56  }
