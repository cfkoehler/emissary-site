122     /**
123      * Tests the sample and if successful provides the description
124      */
125     public String describe(byte[] data) {
126         String desc = describeSelf(data);
127         if (desc == null)
128             return null;
129         StringBuilder sb = new StringBuilder(desc);
130         return escapeBackspace(describeDependents(data, sb, 0));
131     }

133     /**
134      * Private formatting method for escaping backspaces
135      */
136     private static String escapeBackspace(String desc) {
137         String s = "";
138         for (int i = 0; i < desc.length(); i++) {
139             if (desc.charAt(i) == '\\' && (i + 1) < desc.length() && desc.charAt(i + 1) == 'b') {
140                 s = s.substring(0, s.length() - 1);
141                 i++;
142                 continue;
143             }
144             s += desc.charAt(i);
145         }
146         return s;
147     }

158     /**
159      * Private method to format output - mainly for description substitutions
160      */
161     private String format(String desc, byte[] data) {
162 
163         if (!substitute)
164             return desc;
165         Stack<Character> stack = new Stack<>();
166         for (int i = (desc.length() - 1); i >= 0; --i)
167             stack.push(desc.charAt(i));
168         StringBuilder sb = new StringBuilder();
169 
170         while (!stack.empty()) {
171             Character next = stack.pop();
172             if (!stack.empty() && next.charValue() == '%') {
173                 Character subType = stack.pop();
174                 try {
175                     if (dataType == TYPE_STRING) {
176                         if (offset < (data.length - 2)) {
177                             String sub = new String(getElement(data, offset, 1), DEFAULT_CHARSET);
178                             sb.append(sub);
179                         }
180                     } else if (subType.charValue() == 'c' || subType.charValue() == 's') {
181 
182                         byte[] subData = getElement(data, offset, dataTypeLength);
183                         if (subData != null) {
184                             String sub = new String(subData, DEFAULT_CHARSET);
185                             sb.append(sub);
186                         }
187 
188                     } else {
189 
190                         byte[] subData = getElement(data, offset, dataTypeLength);
191                         if (subData != null) {
192                             String sub = MagicMath.byteArrayToString(subData, 10);
193                             sb.append(sub);
194                         }
195                     }
196                 } catch (UnsupportedEncodingException e) {
197                     throw new RuntimeException(e);
198                 }
199                 if (subType.charValue() == 'l' && !stack.empty() && (stack.peek()).charValue() == 'd')
200                     stack.pop();
201                 continue;
202             }
203             sb.append(next.charValue());
204         }
205         return sb.toString();
206     }

208     /**
209      * Tests dependent children
210      */
211     private String describeDependents(byte[] data, StringBuilder sb, int layer) {
212         log.debug("DESCRIBING DEPENDENTS at layer {}", layer);
213         if (dependencies == null || layer >= dependencies.size()) {
214             log.debug("Not enough dependents for layer {}", layer);
215             return sb.toString();
216         }
217 
218         boolean shouldContinue = false;
219         MagicNumber[] dependentItems = dependencies.get(layer);
220         log.debug("Found {} items at layer {}", dependentItems.length, layer);
221         for (int i = 0; i < dependentItems.length; i++) {
222             String s = dependentItems[i].describeSelf(data);
223 
224             if (s != null) {
225                 if (sb.length() > 0) {
226                     sb.append(' ');
227                 }
228                 sb.append(s);
229                 shouldContinue = true;
230             }
231         }
232 
233         if (!shouldContinue)
234             return sb.toString();
235         return describeDependents(data, sb, layer + 1);
236     }

252     /**
253      * Tests this magic number against the given data
254      */
255     public boolean test(byte[] data) {
256         byte[] subject = getElement(data, offset, dataTypeLength);
257         if (subject == null)
258             return false;
259         // printByteSample(subject, "DATA SAMPLE: ");
260         return testNumeric(subject);
261     }

263     /**
264      * Tests numeric byte data only
265      */
266     private boolean testNumeric(byte[] data) {
267         if (substitute)
268             return true;
269         byte[] mValues = value;
270 
271         log.debug("Unary Operator: {}", unaryOperator);
272         // printByteSample(mValues, "MAGIC VALUE: ");
273 
274         int end = mValues.length;
275         switch (unaryOperator) {
276             case MAGICOPERATOR_AND:
277                 for (int i = 0; i < end; i++) {
278                     if (data[i] != mValues[i])
279                         return false;
280                 }
281                 return true;
282             case MAGICOPERATOR_GTHAN:
283                 for (int i = 0; i < end; i++) {
284                     if ((data[i] & 0xFF) < (mValues[i] & 0xFF))
285                         return false;
286                     if (i == end - 1 && data[i] == mValues[i])
287                         return false;
288                 }
289                 return true;
290             case MAGICOPERATOR_LTHAN:
291                 for (int i = 0; i < end; i++) {
292                     if ((data[i] & 0xFF) > (mValues[i] & 0xFF))
293                         return false;
294                     if (i == end - 1 && data[i] == mValues[i])
295                         return false;
296                 }
297                 return true;
298             case MAGICOPERATOR_OR:
299                 for (int i = 0; i < end; i++) {
300                     if (data[i] == mValues[i])
301                         return true;
302                 }
303                 return false;
304             case MAGICOPERATOR_BWAND:
305                 for (int i = 0; i < end; i++) {
306                     if (data[i] != mValues[i])
307                         return false;
308                 }
309                 return true;
310             case MAGICOPERATOR_BWNOT:
311                 for (int i = 0; i < end; i++) {
312                     if (data[i] != mValues[i])
313                         return true;
314                 }
315                 return false;
316             case MAGICOPERATOR_NOT:
317                 for (int i = 0; i < end; i++) {
318                     if (data[i] != mValues[i])
319                         return true;
320                 }
321                 return false;
322             case MAGICOPERATOR_EQUAL_GTHAN:
323                 for (int i = 0; i < end; i++) {
324                     if ((data[i] & 0xFF) < (mValues[i] & 0xFF))
325                         return false;
326                 }
327                 return true;
328             case MAGICOPERATOR_EQUAL_LTHAN:
329                 for (int i = 0; i < end; i++) {
330                     if ((data[i] & 0xFF) > (mValues[i] & 0xFF))
331                         return false;
332                 }
333                 return true;
334             default:
335                 throw new RuntimeException(
336                         "This MagicNumber instance is configured incorrectly. The unary operator is set to an unknown or unconfigured value.");
337 
338         }
339     }

341     /**
342      * Retrieves the data sample
343      */
344     private static byte[] getElement(@Nullable byte[] data, int offset, int length) {
345         if (data == null)
346             return null;
347         if (data.length < (offset + length))
348             return null;
349         // log.info ("SAMPLE STATS - offset: {}, length: {}", offset, length);
350         byte[] subject = new byte[length];
351         for (int i = 0; i < subject.length; i++)
352             subject[i] = data[i + offset];
353         return subject;
354     }

365     /**
366      * Re-creates the string magic number entry for this number only
367      *
368      * @return a String represention of the entry
369      */
370     @Override
371     public String toString() {
372 
373         StringBuilder sb = new StringBuilder();
374         for (int i = 0; i < depth; i++)
375             sb.append('>');
376         if (offsetUnary > 0)
377             sb.append(offsetUnary);
378 
379         if (offset == 0)
380             sb.append("0");
381         else {
382             sb.append(MagicMath.HEX_PREFIX);
383             sb.append(Integer.toHexString(offset));
384         }
385 
386         sb.append('\t');
387         sb.append(MagicNumberFactory.resolveReverseDataType(dataType));
388         if (mask != null && mask.length > 0) {
389             sb.append('&');
390             sb.append(MagicMath.byteArrayToHexString(mask));
391         }
392 
393         sb.append('\t');
394         if (unaryOperator == MAGICOPERATOR_EQUAL_LTHAN)
395             sb.append("<=");
396         else if (unaryOperator == MAGICOPERATOR_EQUAL_GTHAN)
397             sb.append(">=");
398         else
399             sb.append(unaryOperator);
400         sb.append(MagicMath.byteArrayToHexString(value));
401 
402         sb.append('\t');
403         sb.append(description);
404         return sb.toString();
405     }

407     /**
408      * Private method to create the string plus continuations
409      */
410     private String toString(StringBuilder sbuf, int depth) {
411         StringBuilder sb = sbuf;
412         int d = depth;
413         if (sb == null)
414             sb = new StringBuilder(description);
415         if (dependencies == null || d >= dependencies.size())
416             return sb.toString();
417         MagicNumber[] dependentItems = dependencies.get(d);
418         for (int i = 0; i < dependentItems.length; i++) {
419             sb.append('\n');
420             sb.append(dependentItems[i].toString());
421         }
422         return toString(sb, d + 1);
423     }
