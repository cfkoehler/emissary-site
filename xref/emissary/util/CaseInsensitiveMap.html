3   import java.util.HashMap;
4   import java.util.Map;

6   /**
7    * Be a HashMap that doesn't care about the case of the keys
8    */
9   public class CaseInsensitiveMap<K, V> extends HashMap<K, V> {
10      // serializable
11      static final long serialVersionUID = -8692638820306464417L;
12  
13      // For remapping string keys to through a case-insensitive layer
14      private Map<String, String> remap = new HashMap<>();
15  
16      /**
17       * Create one
18       */
19      public CaseInsensitiveMap() {
20          super();
21      }
22  
23      /**
24       * Create one with capacity
25       */
26      public CaseInsensitiveMap(int initialCapacity) {
27          super(initialCapacity);
28      }
29  
30      /**
31       * Create one with capacity and load factor
32       */
33      public CaseInsensitiveMap(int initialCapacity, float loadFactor) {
34          super(initialCapacity, loadFactor);
35      }
36  
37      /**
38       * Create one from another map
39       */
40      public CaseInsensitiveMap(Map<? extends K, ? extends V> map) {
41          super(map.size());
42          putAll(map);
43      }
44  
45      /**
46       * Put an element into the map If it is a string, put it in and also add it to our case remapping map
47       */
48      @Override
49      // @SuppressWarnings("unchecked")
50      public V put(K key, V value) {
51          if (key instanceof String) {
52              String uckey = ((String) key).toLowerCase();
53              if (remap.containsKey(uckey)) {
54                  this.remove(uckey);
55              }
56              remap.put(uckey, (String) key);
57          }
58  
59          return super.put(key, value);
60      }
61  
62  
63      /**
64       * Get an element from the map If the key is a string check our remap first
65       */
66      @Override
67      public V get(Object key) {
68          Object realkey = key;
69  
70          if (key instanceof String) {
71              String strkey = remap.get(((String) key).toLowerCase());
72              if (strkey != null) {
73                  realkey = strkey;
74              }
75          }
76  
77          return super.get(realkey);
78      }
79  
80      /**
81       * Put all
82       */
83      @Override
84      public void putAll(Map<? extends K, ? extends V> t) {
85          for (Map.Entry<? extends K, ? extends V> entry : t.entrySet()) {
86              this.put(entry.getKey(), entry.getValue());
87          }
88      }
89  
90      /**
91       * Contains key
92       */
93      @Override
94      public boolean containsKey(Object key) {
95          Object realkey = key;
96  
97          if (key instanceof String) {
98              String strkey = remap.get(((String) key).toLowerCase());
99              if (strkey != null) {
100                 realkey = strkey;
101             }
102         }
103 
104         return super.containsKey(realkey);
105     }
106 
107     /**
108      * Remove a mapping
109      */
110     @Override
111     public V remove(Object key) {
112         Object realkey = key;
113 
114         if (key instanceof String) {
115             String uckey = ((String) key).toLowerCase();
116             String strkey = remap.get(uckey);
117             if (strkey != null) {
118                 realkey = strkey;
119             }
120 
121             remap.remove(uckey);
122         }
123 
124         return super.remove(realkey);
125     }
126 
127 }
