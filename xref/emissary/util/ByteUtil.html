114     /**
115      *
116      * @param b a byte array
117      * @param pos a position in the byte array
118      * @return true if byte at pos in array b is a control or whitespace byte
119      */
120     public static boolean isControlOrWhiteSpace(byte[] b, int pos) {
121         if (b[pos] == Ascii_DEL || b[pos] <= Ascii_SP) {
122             return true;
123         }
124         if (b[pos] == Ascii_b && pos > 0 && b[pos - 1] == Ascii_ESC) {
125             return true;
126         }
127 
128         // Check if the current pos is the first byte in a UTF-8 C1
129         // control character (U+0080..U+009f).
130         final int curr = b[pos] & 0xff;
131         final int next = (pos < (b.length - 1)) ? (b[pos + 1] & 0xff) : -1;
132         if ((curr == 0xc2) && (next >= 0x80) && (next <= 0x9f)) {
133             return true;
134         }
135 
136         // Check if the current pos is the second byte in a UTF-8 C1
137         // control character (U+0080..U+009f).
138         final int prev = (pos > 0) ? (b[pos - 1] & 0xff) : -1;
139         return (prev == 0xc2) && (curr >= 0x80) && (curr <= 0x9f);
140     }

142     /**
143      * Glue two byte arrays together into one
144      * 
145      * @param a the first byte array
146      * @param b the second byte array
147      * @return the whole
148      */
149     public static byte[] glue(@Nullable byte[] a, @Nullable byte[] b) {
150         if (a == null) {
151             return b;
152         }
153         if (b == null) {
154             return a;
155         }
156         return glue(a, 0, a.length - 1, b, 0, b.length - 1);
157     }

159     /**
160      * Glue three byte arrays together into one
161      * 
162      * @param a the first byte array
163      * @param b the second byte array
164      * @param c the third byte array
165      * @return the whole
166      */
167     public static byte[] glue(@Nullable byte[] a, @Nullable byte[] b, @Nullable byte[] c) {
168         if (a == null) {
169             return glue(b, c);
170         }
171         if (b == null) {
172             return glue(a, c);
173         }
174         if (c == null) {
175             return glue(a, b);
176         }
177         return glue(a, 0, a.length - 1, b, 0, b.length - 1, c, 0, c.length - 1);
178     }

180     /**
181      * Glue two byte arrays together into one
182      * 
183      * @param a the first byte array
184      * @param astart starting position in a
185      * @param aend ending position in a
186      * @param b the second byte array
187      * @param bstart starting position in b
188      * @param bend ending position in b
189      * @return the whole
190      */
191     public static byte[] glue(byte[] a, int astart, int aend, byte[] b, int bstart, int bend) {
192         int alen = aend - astart + 1;
193         int blen = bend - bstart + 1;
194 
195         byte[] rslt = new byte[alen + blen];
196         System.arraycopy(a, astart, rslt, 0, alen);
197         System.arraycopy(b, bstart, rslt, alen, blen);
198         return rslt;
199     }

201     /**
202      * Glue three byte arrays together into one
203      * 
204      * @param a the first byte array
205      * @param astart starting position in a
206      * @param aend ending position in a
207      * @param b the second byte array
208      * @param bstart starting position in b
209      * @param bend ending position in b
210      * @param c the third byte array
211      * @param cstart starting position in c
212      * @param cend ending position in c
213      * @return the whole
214      */
215     public static byte[] glue(byte[] a, int astart, int aend, byte[] b, int bstart, int bend, byte[] c, int cstart, int cend) {
216         int alen = aend - astart + 1;
217         int blen = bend - bstart + 1;
218         int clen = cend - cstart + 1;
219 
220         byte[] rslt = new byte[alen + blen + clen];
221         System.arraycopy(a, astart, rslt, 0, alen);
222         System.arraycopy(b, bstart, rslt, alen, blen);
223         System.arraycopy(c, cstart, rslt, alen + blen, clen);
224         return rslt;
225     }

227     /**
228      * Split a byte array at the specified position
229      * 
230      * @param a the byte array
231      * @param pos the split position (a[pos] goes to the second part)
232      */
233     public static List<byte[]> split(@Nullable byte[] a, int pos) {
234         List<byte[]> list = new ArrayList<>();
235         if (a != null && pos > 0 && pos <= a.length) {
236             byte[] part1 = new byte[pos];
237             byte[] part2 = new byte[a.length - pos];
238             System.arraycopy(a, 0, part1, 0, pos);
239             System.arraycopy(a, pos, part2, 0, part2.length);
240             list.add(part1);
241             list.add(part2);
242         } else {
243             // Just give back the original
244             list.add(a);
245         }
246         return list;
247     }

249     /**
250      * Given a byte-array and a start offset, return a string of the bytes between the start position and a carriage return
251      * byte. In essence, this is grabbing a line of input where the byte array is composed of several lines of input.
252      * 
253      * @param data The byte array of input data.
254      * @param pos The initial start offset.
255      * @return A string created from the bytes found from the start offset to the carriage return byte.
256      */
257     public static String grabLine(byte[] data, int pos) {
258         String ret = null;
259         int eolnPos = -1;
260         for (int i = pos; i < data.length; i++) {
261             if (data[i] == '\n') {
262                 eolnPos = i;
263                 break;
264             }
265         }
266         if (eolnPos != -1) {
267             // String up to the found \n pos
268             ret = new String(data, pos, eolnPos - pos + 1);
269         } else {
270             // String to end of buffer
271             ret = new String(data, pos, data.length - pos);
272         }
273         return ret;
274     }
