16  /**
17   * The byte tokenizer class allows an application to break a byte buffer into tokens. This was modified from the
18   * java.util.StringTokenizer implementation. Note that all characters in the deliminter set are considered to be
19   * characters in the range 0 - 255. In other words the ISO8859-1 encoding is used to match the delimiters to the byte
20   * array.
21   */
22  public class ByteTokenizer implements Enumeration<String> {
23      private int currentPosition;
24      private int newPosition;
25      private int maxPosition;
26      private byte[] data;
27      private String delimiters;
28      private boolean retDelims;
29      private boolean delimsChanged;
30      private String encoding;
31  
32      private static final Logger logger = LoggerFactory.getLogger(ByteTokenizer.class);
33  
34      /**
35       * maxDelimChar stores the value of the delimiter character with the highest value. It is used to optimize the detection
36       * of delimiter characters.
37       */
38      private char maxDelimChar;
39  
40      /**
41       * Set maxDelimChar to the highest char in the delimiter set.
42       */
43      private void setMaxDelimChar() {
44          if (delimiters == null) {
45              maxDelimChar = 0;
46              return;
47          }
48  
49          char m = 0;
50          for (int i = 0; i < delimiters.length(); i++) {
51              char c = delimiters.charAt(i);
52              if (m < c) {
53                  m = c;
54              }
55          }
56          maxDelimChar = m;
57      }
58  
59      /**
60       * Constructs a byte tokenizer for the specified byte array. All characters in the <code>delim</code> argument are the
61       * delimiters for separating tokens. Characters must be in the range of 0 - 255.
62       * <p>
63       * If the <code>returnDelims</code> flag is <code>true</code>, then the delimiter characters are also returned as
64       * tokens. Each delimiter is returned as a byte[] or String of length one. If the flag is <code>false</code>, the
65       * delimiter characters are skipped and only serve as separators between tokens.
66       *
67       * @param bytes a byte array to be parsed.
68       * @param start the first byte in the array
69       * @param len the number of bytes to parse
70       * @param delim the delimiters.
71       * @param returnDelims flag indicating whether to return the delimiters as tokens.
72       */
73      public ByteTokenizer(byte[] bytes, int start, int len, String delim, boolean returnDelims) {
74          currentPosition = start;
75          newPosition = -1;
76          delimsChanged = false;
77          data = bytes;
78          maxPosition = start + len;
79          delimiters = delim;
80          retDelims = returnDelims;
81          setMaxDelimChar();
82      }

84      /**
85       * Constructs a byte tokenizer for the specified byte array using the specified encoding.
86       *
87       * @param bytes a byte array to be parsed.
88       * @param start the first byte in the array
89       * @param len the number of bytes to parse
90       * @param delim the delimiters.
91       * @param returnDelims flag indicating whether to return the delimiters as tokens.
92       * @param encoding the encoding for which to return the bytes.
93       * @exception UnsupportedEncodingException thrown if the supplied encoding is unsupported.
94       */
95      public ByteTokenizer(byte[] bytes, int start, int len, String delim, boolean returnDelims, String encoding) throws UnsupportedEncodingException {
96          this(bytes, start, len, delim, returnDelims);
97          try {
98              Charset c = Charset.forName(encoding);
99              logger.debug("Loaded charset {}", c);
100         } catch (Exception ex) {
101             throw new UnsupportedEncodingException("No support for " + encoding);
102         }
103         this.encoding = encoding;
104     }

211     /**
212      * Skips delimiters starting from the specified position. If retDelims is false, returns the index of the first
213      * non-delimiter character at or after startPos. If retDelims is true, startPos is returned.
214      */
215     private int skipDelimiters(int startPos) {
216         if (delimiters == null) {
217             throw new NullPointerException();
218         }
219 
220         int position = startPos;
221         while (!retDelims && position < maxPosition) {
222             char c = (char) (0xFF & data[position]);
223             if ((c > maxDelimChar) || (delimiters.indexOf(c) < 0)) {
224                 break;
225             }
226             position++;
227         }
228         return position;
229     }
230 
231     /**
232      * Skips ahead from startPos and returns the index of the next delimiter character encountered, or maxPosition if no
233      * such delimiter is found.
234      */
235     private int scanToken(int startPos) {
236         int position = startPos;
237         while (position < maxPosition) {
238             char c = (char) (0xFF & data[position]);
239             if ((c <= maxDelimChar) && (delimiters.indexOf(c) >= 0)) {
240                 break;
241             }
242             position++;
243         }
244         if (retDelims && (startPos == position)) {
245             char c = (char) (0xFF & data[position]);
246             if ((c <= maxDelimChar) && (delimiters.indexOf(c) >= 0)) {
247                 position++;
248             }
249         }
250         return position;
251     }

253     /**
254      * Tests if there are more tokens available from this tokenizer's string. If this method returns <code>true</code>, then
255      * a subsequent call to <code>nextToken</code> with no argument will successfully return a token.
256      *
257      * @return <code>true</code> if and only if there is at least one token in the string after the current position;
258      *         <code>false</code> otherwise.
259      */
260     public boolean hasMoreTokens() {
261         /*
262          * Temporary store this position and use it in the following nextToken() method only if the delimiters haven't been
263          * changed in that nextToken() invocation.
264          */
265         newPosition = skipDelimiters(currentPosition);
266         return (newPosition < maxPosition);
267     }
268 
269     /**
270      * Returns the next token from this string tokenizer.
271      *
272      * @return the next token from this string tokenizer.
273      * @exception NoSuchElementException if there are no more tokens in this tokenizer's string.
274      */
275     public String nextToken() {
276         /*
277          * If next position already computed in hasMoreElements() and delimiters have changed between the computation and this
278          * invocation, then use the computed value.
279          */
280 
281         currentPosition = (newPosition >= 0 && !delimsChanged) ? newPosition : skipDelimiters(currentPosition);
282 
283         /* Reset these anyway */
284         delimsChanged = false;
285         newPosition = -1;
286 
287         if (currentPosition >= maxPosition) {
288             throw new NoSuchElementException();
289         }
290         int start = currentPosition;
291         currentPosition = scanToken(currentPosition);
292 
293         String token = null;
294         try {
295             if (encoding != null) {
296                 token = new String(data, start, currentPosition - start, encoding);
297             } else {
298                 token = new String(data, start, currentPosition - start);
299             }
300         } catch (UnsupportedEncodingException uee) {
301             // cannot happen...we already verified in constructer
302         }
303         return token;
304     }

306     /**
307      * Returns the next token in this string tokenizer's string. First, the set of characters considered to be delimiters by
308      * this <code>ByteTokenizer</code> object is changed to be the characters in the string <code>delim</code>. Then the
309      * next token in the string after the current position is returned. The current position is advanced beyond the
310      * recognized token. The new delimiter set remains the default after this call.
311      *
312      * @param delim the new delimiters.
313      * @return the next token, after switching to the new delimiter set.
314      * @exception NoSuchElementException if there are no more tokens in this tokenizer's string.
315      */
316     public String nextToken(String delim) {
317         delimiters = delim;
318 
319         /* delimiter string specified, so set the appropriate flag. */
320         delimsChanged = true;
321 
322         setMaxDelimChar();
323         return nextToken();
324     }

328      * <code>Enumeration</code> interface.
329      *
330      * @return <code>true</code> if there are more tokens; <code>false</code> otherwise.
331      * @see java.util.Enumeration

334     @Override
335     public boolean hasMoreElements() {
336         return hasMoreTokens();

339     /**
340      * Returns the same value as the <code>nextToken</code> method, except that its declared return value is
341      * <code>Object</code> rather than <code>String</code>. It exists so that this class can implement the
342      * <code>Enumeration</code> interface.
343      *
344      * @return the next token in the string.

347      * @see ByteTokenizer#nextToken()
348      */
349     @Override
350     public String nextElement() {
351         return nextToken();
352     }

354     /**
355      * Calculates the number of times that this tokenizer's <code>nextToken</code> method can be called before it generates
356      * an exception. The current position is not advanced.
357      *
358      * @return the number of tokens remaining in the string using the current delimiter set.
359      * @see ByteTokenizer#nextToken()
360      */
361     public int countTokens() {
362         int count = 0;
363         int currpos = currentPosition;
364         while (currpos < maxPosition) {
365             currpos = skipDelimiters(currpos);
366             if (currpos >= maxPosition) {
367                 break;
368             }
369             currpos = scanToken(currpos);
370             count++;
371         }
372         return count;
373     }
