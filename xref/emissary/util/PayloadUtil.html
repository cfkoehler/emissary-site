29      /**
30       * Try really hard to get a meaningful name for a payload object
31       */
32      public static String getName(final Object o) {
33          String payloadName = o.getClass().getName();
34          if (o instanceof IBaseDataObject) {
35              payloadName = ((IBaseDataObject) o).shortName();
36          } else if (o instanceof Collection) {
37              final Iterator<?> pi = ((Collection<?>) o).iterator();
38              if (pi.hasNext()) {
39                  payloadName = ((IBaseDataObject) pi.next()).shortName() + "(" + ((Collection<?>) o).size() + ")";
40              }
41          }
42          return payloadName;
43      }

55      /**
56       * Generate a string about the payload object
57       * 
58       * @param payload the payload to describe
59       */
60      public static String getPayloadDisplayString(final IBaseDataObject payload) {
61          final StringBuilder sb = new StringBuilder();
62          final List<TransformHistory.History> th = payload.getTransformHistory().getHistory();
63          final String fileName = payload.getFilename();
64          final List<String> currentForms = payload.getAllCurrentForms();
65          final Date creationTimestamp = payload.getCreationTimestamp();
66  
67          sb.append("\n").append("filename: ").append(fileName).append("\n").append("   creationTimestamp: ").append(creationTimestamp).append("\n")
68                  .append("   currentForms: ").append(currentForms).append("\n").append("   filetype: ").append(payload.getFileType()).append("\n")
69                  .append("   transform history (").append(th.size()).append(") :").append("\n");
70          for (final TransformHistory.History h : th) {
71              sb.append("     ");
72              if (h.wasCoordinated()) {
73                  sb.append("  ");
74              }
75              sb.append(h.getKey()).append("\n");
76          }
77          return sb.toString();
78      }

80      /**
81       * Generate a one-line string about the payload object
82       * 
83       * @param payload the payload to describe
84       */
85      public static String getPayloadOneLineString(final IBaseDataObject payload) {
86          final StringBuilder sb = new StringBuilder();
87          final String fn = payload.getFilename();
88          final int attPos = fn.indexOf(Family.SEP);
89          if (attPos != -1) {
90              sb.append(fn.substring(attPos + 1)).append(" ");
91          }
92          final List<String> th = payload.transformHistory(true);
93          String prev = "";
94          for (final String h : th) {
95              final int pos = h.indexOf(".");
96              if (pos > 0) {
97                  final String prefix = h.substring(0, pos);
98                  if (!prev.equals(prefix)) {
99                      if (prev.length() != 0) {
100                         sb.append(",");
101                     }
102                     sb.append(prefix);
103                     prev = prefix;
104                 }
105             }
106         }
107         sb.append(">>").append(payload.getAllCurrentForms());
108         sb.append("//").append(payload.getFileType());
109         sb.append("//").append(payload.getCreationTimestamp());
110         return sb.toString();
111     }

113     /**
114      * Turn the payload into an xml jdom document
115      * 
116      * @param d the payload
117      */
118     public static Document toXml(final IBaseDataObject d) {
119         final Element root = new Element("payload");
120         root.addContent(JDOMUtil.protectedElement("name", d.getFilename()));
121         final Element cf = new Element("current-forms");
122         for (final String c : d.getAllCurrentForms()) {
123             cf.addContent(JDOMUtil.simpleElement("current-form", c));
124         }
125         root.addContent(cf);
126         root.addContent(JDOMUtil.simpleElement("encoding", d.getFontEncoding()));
127         root.addContent(JDOMUtil.simpleElement("filetype", d.getFileType()));
128         root.addContent(JDOMUtil.simpleElement("classification", d.getClassification()));
129         final Element th = new Element("transform-history");
130         for (final String s : d.transformHistory(true)) {
131             th.addContent(JDOMUtil.simpleElement("itinerary-step", s));
132         }
133         root.addContent(th);
134         if (d.getProcessingError() != null) {
135             root.addContent(JDOMUtil.simpleElement("processing-error", d.getProcessingError()));
136         }
137         final Element meta = new Element("metadata");
138         for (final String key : d.getParameters().keySet()) {
139             final Element m = JDOMUtil.protectedElement("param", d.getStringParameter(key));
140             m.setAttribute("name", key);
141             meta.addContent(m);
142         }
143         root.addContent(meta);
144 
145         if (d.header() != null) {
146             root.addContent(JDOMUtil.protectedElement("header", d.header()));
147         }
148         if (d.dataLength() > 0) {
149             root.addContent(JDOMUtil.protectedElement("data", d.data()));
150         }
151         if (d.footer() != null) {
152             root.addContent(JDOMUtil.protectedElement("footer", d.footer()));
153         }
154 
155         // Alt views
156         if (d.getNumAlternateViews() > 0) {
157             final Element views = new Element("views");
158             for (final String av : d.getAlternateViewNames()) {
159                 final Element v = JDOMUtil.protectedElement("view", d.getAlternateView(av));
160                 v.setAttribute("name", av);
161                 views.addContent(v);
162             }
163             root.addContent(views);
164         }
165 
166         logger.debug("Produced xml document for " + d.shortName());
167         return new Document(root);
168     }

179     /**
180      * Turn a list of payload into an xml jdom ocument
181      * 
182      * @param list the payload list
183      */
184     public static Document toXml(final List<IBaseDataObject> list) {
185         final Element root = new Element("payload-list");
186         for (final IBaseDataObject d : list) {
187             final Document doc = toXml(d);
188             root.addContent(doc.detachRootElement());
189             logger.debug("Adding xml content for " + d.shortName() + " to document");
190         }
191         return new Document(root);
192     }

203     /**
204      * Print formatted metadata key:value pairs
205      */
206     private static final String SEP = ": ";
207 
208     public static String printFormattedMetadata(final IBaseDataObject payload) {
209         final StringBuilder out = new StringBuilder();
210         out.append(LS);
211         for (final Map.Entry<String, Collection<Object>> entry : payload.getParameters().entrySet()) {
212             out.append(entry.getKey()).append(SEP).append(entry.getValue()).append(LS);
213         }
214         return out.toString();
215     }

217     /**
218      * Checks whether the form complies with form rules established by a regex
219      *
220      * Approved forms can contain alpha-numerics, '-', or '_'
221      *
222      * @param form The form to be tested
223      * @return Whether the form is Emissary compliant
224      */
225     public static boolean isValidForm(String form) {
226         return validFormRegex.matcher(form).find();
227     }
