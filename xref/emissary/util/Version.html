26      /**
27       * Read hardwired config info. Cannot be overridden using the normal emissary.config.pkg or emissary.config.dir methods.
28       * Does not create a Configurator due to logging restrictions on this method.
29       */
30      private void readHardWiredConfigInfo() {
31          InputStream rstream = null;
32          try {
33              // This configurator is not overridable by
34              // the normal config.pkg or config.dir mechanisms.
35              // It also cannot call any method in ConfigUtil that
36              // will end up calling a logger.* method, since we
37              // are here producing the header for the log file
38              String rez = Version.class.getName().replace('.', '/') + ConfigUtil.CONFIG_FILE_ENDING;
39              // System.out.println("Reading " + rez);
40              rstream = new ResourceReader().getResourceAsStream(rez);
41              if (rstream != null) {
42                  byte[] data = new byte[rstream.available()];
43                  rstream.read(data, 0, data.length);
44                  String sdata = new String(data);
45                  String[] lines = sdata.split("[\r\n]+");
46                  if (lines != null) {
47                      // System.out.println(" Got " + lines.length + " lines");
48                      for (int i = 0; i < lines.length; i++) {
49                          // System.out.println("Got line " + lines[i]);
50                          if (lines[i].startsWith("emissary_version")) {
51                              version = getVal(lines[i]);
52                          } else if (lines[i].startsWith("emissary_build")) {
53                              timestamp = getVal(lines[i]);
54                          }
55                      }
56                  }
57                  // else
58                  // {
59                  // System.out.println("No lines to read");
60                  // }
61              }
62          } catch (IOException iox) {
63              // System.out.println("Bad read: " + iox);
64          } finally {
65              if (rstream != null) {
66                  try {
67                      rstream.close();
68                  } catch (IOException ignore) {
69                  }
70              }
