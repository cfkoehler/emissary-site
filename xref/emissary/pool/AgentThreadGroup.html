3   import org.slf4j.Logger;
4   import org.slf4j.LoggerFactory;
5   
6   /**
7    * Provide a ThreadGroup and an uncaught exception handler for our agents to be a part of
8    */
9   public class AgentThreadGroup extends ThreadGroup {
10  
11      private static final Logger logger = LoggerFactory.getLogger(AgentThreadGroup.class);
12  
13      public AgentThreadGroup(String s) {
14          super(s);
15      }
16  
17      @Override
18      public void uncaughtException(Thread thread, Throwable throwable) {
19          if (throwable instanceof java.lang.ThreadDeath) {
20              logger.error("Fatal ThreadDeath on {} had ThreadDeath exception ", thread.getName(), throwable);
21              throw (java.lang.ThreadDeath) throwable;
22          }
23          logger.error("Fatal Thread Error on {} had an uncaught exception ", thread.getName(), throwable);
24      }
25  }
