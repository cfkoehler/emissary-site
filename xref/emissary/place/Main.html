430                 fcf = new File(cf);
431                 if (fcf.exists()) {
432                     // Save it if long name does not exist but short one does
433                     setConfigLocation(classOnlyName + ConfigUtil.CONFIG_FILE_ENDING);
434                 }
435             }
436         }
437 
438         boolean pseudoNodeCreated = false;
439         try {

441             if (!node.isValid()) {

445                 pseudoNodeCreated = true;

466     /**
467      * Run the main method in normal testing mode using the standard or configured options and taking the remaining
468      * arguments as data files or directories of data files to be processed
469      */
470     public void run() {
471         parseArguments();
472 
473         // Create the pool of workers
474         logger.debug("Setting up " + getThreads() + " worker threads");
475         for (int i = 0; i < getThreads(); i++) {
476             Worker w = new Worker();
477             new Thread(w, "MainWorker-" + i).start();
478             workers.add(w);
479         }
480 
481         if (isp != null) {
482             do {
483                 // Process the data, loading it into the work queue
484                 processMainData(fileArgs, null);
485 
486                 // Wait for the work queue to be empty
487                 int size = -1;
488                 synchronized (workQueue) {
489                     size = workQueue.size();
490                 }
491                 while (size > 0) {
492                     int workQueueSize = -1;
493                     synchronized (workQueue) {
494                         workQueueSize = workQueue.size();
495                     }
496                     if (workQueueSize < size) {
497                         size = workQueueSize;
498                         logger.debug("Waiting for work queue " + size + " remaining");
499                     }
500                     try {
501                         Thread.sleep(100);
502                     } catch (InterruptedException ignore) {
503                         Thread.currentThread().interrupt();
504                     }
505                 }
506             } while (loopOnInput);
507         }
508 
509         // Shutdown the workers
510         logger.debug("Stopping " + workers.size() + " workers");
511         for (int i = 0; i < getThreads(); i++) {
512             workers.get(i).stop();
513         }
514 
515         synchronized (workQueue) {
516             workQueue.notifyAll();
517         }
518 
519         // Wait for all workers to actually stop
520         for (int i = 0; i < getThreads(); i++) {
521             Worker w = workers.get(i);
522             while (!w.isIdle()) {
523                 try {
524                     Thread.sleep(10);
525                 } catch (InterruptedException ignore) {
526                     Thread.currentThread().interrupt();
527                 }
528             }
529         }
530     }

552     /**
553      * Parse the command line arguments
554      */
555     public void parseArguments() {
556         CommandLineParser parser = new PosixParser();
557         CommandLine cmd = null;
558         try {
559             cmd = parser.parse(options, args);
560             fileArgs = cmd.getArgs();
561             if (fileArgs != null) {
562                 logger.debug("Parsed arguments and have " + fileArgs.length + " file args remaining");
563             }
564         } catch (ParseException ex) {
565             // oops, something went wrong
566             logger.error(placeClass + ": Parsing failed.  Reason: " + ex.getMessage());
567             printUsage();
568             return;
569         }
570 
571         boolean shouldContinue = preArgumentsHook(cmd);
572 
573         if (shouldContinue) {
574             processParsedArguments(cmd);
575             postArgumentsHook(cmd);
576         }
577 
578     }

580     protected void processParsedArguments(CommandLine cmd) {
581         // Take care of the help option
582         if (cmd.hasOption("h")) {
583             printUsage();
584             return;
585         }
586 
587         // Set boolean flags
588         setSilent(cmd.hasOption("s"));
589         setVerbose(cmd.hasOption("v"));
590         setRecursive(cmd.hasOption("R"));
591         setLoopOnInput(cmd.hasOption("i"));
592         setOneLiner(cmd.hasOption("1"));
593         setRunWithoutContext(cmd.hasOption("X"));
594 
595         // Override config if specified
596         if (cmd.hasOption("c")) {
597             setConfigLocation(cmd.getOptionValue("c"));
598         }
599 
600         // Set up a parser if one is specified
601         if (cmd.hasOption("P")) {
602             setParserName(cmd.getOptionValue("P"));
603         }
604 
605         // Override current form if specified
606         if (cmd.hasOption("t")) {
607             setCurrentForm(cmd.getOptionValue("t"));
608         }
609 
610         // Set up number of threads to use
611         if (cmd.hasOption("T")) {
612             logger.debug("Thread option is " + cmd.getOptionValue("T"));
613             setThreads(cmd.getOptionValue("T"));
614         }
615 
616         // save parameter key=value pairs to set
617         if (cmd.hasOption("p")) {
618             setParameters(cmd.getOptionValues("p"));
619         }
620 
621         // Save regex's for looking at metadata
622         setMetaPatterns(cmd.getOptionValues("m"));
623 
624         // Save alt view names to print in a set
625         String[] altViews = cmd.getOptionValues("a");
626         for (int i = 0; altViews != null && i < altViews.length; i++) {
627             viewsToPrint.add(altViews[i]);
628         }
629 
630         // Do some of the things that the normal context initializer does
631         boolean pseudoNodeCreated = false;
632         if (!isRunWithoutContext()) {
633             try {
634                 EmissaryNodemissaryNode">EmissaryNode node = new EmissaryNode();
635                 if (!node.isValid()) {
636                     System.setProperty(EmissaryNode.NODE_NAME_PROPERTY, "localhost");
637                     System.setProperty(EmissaryNode.NODE_PORT_PROPERTY, "0000");
638                     node = new EmissaryNode();
639                     pseudoNodeCreated = true;
640                 }
641                 new DirectoryPlace("EMISSARY_DIRECTORY_SERVICES.STUDY.DIRECTORY_PLACE.http://" + node.getNodeName() + ":" + node.getNodePort()
642                         + "/DirectoryPlace", node);
643             } catch (IOException iox) {
644                 logger.debug("Could not create standalone directory", iox);
645             } finally {
646                 if (pseudoNodeCreated) {
647                     System.clearProperty(EmissaryNode.NODE_NAME_PROPERTY);
648                     System.clearProperty(EmissaryNode.NODE_PORT_PROPERTY);
649                 }
650             }
651         }
652 
653         // Create the place
654         if (!createPlace()) {
655             logger.warn("Unable to run main processing, no place created");
656             return;
657         }
658 
659         // Grab the base output directory for split
660         if (cmd.hasOption("d")) {
661             setBaseOutputDir(cmd.getOptionValue("d"));
662         }
663 
664         // Turn on splitting of output if requested
665         if (cmd.hasOption("S")) {
666             setSplitOutput(true);
667         }
668 
669 
670         // Redirect the output stream if desired
671         if (cmd.hasOption("o")) {
672             try {
673                 setOutputStream(cmd.getOptionValue("o"));
674             } catch (IOException iox) {
675                 logger.error("Cannot redirect output to " + cmd.getOptionValue("o") + ": " + iox.getMessage());
676             }
677 
678         }
679     }

699     /**
700      * Set parameters on the new DataObject. Format of pattern is key=value.
701      *
702      * @param keyval the params to use
703      */
704     public void setParameters(String[] keyval) {
705         for (String p : keyval) {
706             if (p.indexOf("=") > 0) {
707                 // limit to two, so key=val1=val2 just turns into
708                 // "key" = "val1=val2"
709                 String[] kv = p.split("=", 2);
710                 params.put(kv[0], kv[1]);
711             } else {
712                 logger.debug("param format must be key=value, skipping  " + p);
713             }
714         }
715     }

817     /**
818      * Process the file arguments from the command-line or recursed directory
819      *
820      * @param args the remaining (non-option) arguments
821      * @param path the path we are operating in, for appending when doing file recursion
822      */
823     public void processMainData(String[] args, @Nullable String path) {
824         for (int i = 0; i < args.length; i++) {
825             String spath = (path != null ? (path + "/") : "") + args[i];
826             File f = new File(spath);
827             if (f.isDirectory()) {
828                 if (isRecursive()) {
829                     String[] files = f.list();
830                     // recursion alert
831                     logger.debug("Recursing into directory " + spath);
832                     processMainData(files, spath);
833                     logger.debug("Leaving recursed directory" + spath);
834                     // end recursive call
835                 } else {
836                     outStream.println(spath + ": DIRECTORY");
837                 }
838                 continue;
839             }
840 
841             if (!f.canRead()) {
842                 outStream.println(spath + ": UNREADABLE");
843                 continue;
844             }
845 
846             // Process the file
847             processFile(spath);
848         }
849     }

851     /**
852      * Process the single named file
853      */
854     public void processFile(String spath) {
855         // This could be expanded to allow configuration to choose
856         // to use a RAF or NIO parser instead of forcing a byte array
857         // parser
858         byte[] content = Executrix.readDataFromFile(spath);
859         if (content == null) {
860             outStream.println(spath + ": UNREADABLE");
861             return;
862         }
863 
864         SessionParser sp = null;
865         try {
866             sp = createParser(content);
867         } catch (Throwable t) {
868             logger.error("Cannot create parser " + getParserName() + ": " + t);
869             sp = new SimpleParser(content);
870         }
871 
872         try {
873             SessionProducernProducer">SessionProducer source = new SessionProducer(sp, currentForm);
874             int count = 1;
875             while (true) {
876                 IBaseDataObject payload = source.getNextSession(spath + (count > 1 ? ("-" + count) : ""));
877                 kff.hash(payload);
878 
879                 // Set up filetype if non-default
880                 if (!emissary.core.Form.UNKNOWN.equals(currentForm)) {
881                     payload.setFileType(currentForm);
882                 }
883 
884                 // add command line metadata before processing
885                 payload.setParameters(params);
886 
887                 // Push this payload onto a queue for the Worker consumers to pull from
888                 queuePayload(payload);
889 
890                 if (count % numThreads == 0) {
891                     Thread.yield();
892                 }
893 
894                 count++;
895             }
896         } catch (ParserEOFException eof) {
897             // expected at end of file
898         } catch (ParserException ex) {
899             logger.error("File " + spath + " cannot be parsed by " + getParserName() + ": " + ex);
900         }
901     }

913     /**
914      * Called by the Worker consumer to process the given payload
915      */
916     public void processPayload(IBaseDataObject payload) {
917         List<IBaseDataObject> attachments = new ArrayList<>();
918         MDC.put(MDCConstants.SHORT_NAME, payload.shortName());
919         try {
920             boolean shouldContinue = preProcessHook(payload, attachments);
921             if (shouldContinue) {
922                 attachments = isp.agentProcessHeavyDuty(payload);
923             }
924             postProcessHook(payload, attachments);
925         } catch (Throwable pex) {
926             payload.replaceCurrentForm(Form.ERROR);
927             attachments = Collections.emptyList();
928             postProcessErrorHook(payload);
929         } finally {
930             MDC.remove(MDCConstants.SHORT_NAME);
931         }
932 
933         if (!isSilent()) {
934             boolean shouldPrint = prePrintHook(payload, attachments);
935 
936             if (shouldPrint) {
937                 if (isOneLiner()) {
938                     printPayloadOneLiner(payload, attachments.size() + payload.getExtractedRecordCount());
939                 } else {
940                     printPayload(payload);
941 
942                     for (IBaseDataObject att : attachments) {
943                         printPayload(att);
944                     }
945 
946                     if (payload.hasExtractedRecords()) {
947                         for (IBaseDataObject rec : payload.getExtractedRecords()) {
948                             printPayload(rec);
949                         }
950                     }
951                 }
952             }
953 
954             postPrintHook(payload, attachments);
955         }
956 
957         // Split output into files if specified
958         // This is independent of printStream output
959         // printing, isSlient, shouldPrint, etc.
960         if (isSplitOutput()) {
961             boolean shoudSplit = preSplitHook(payload, attachments);
962             if (shoudSplit) {
963                 handleSplitOutput(payload, attachments);
964             }
965             postSplitHook(payload, attachments);
966         }
967     }

984      * Print out stuff related to a command-line processed payload
985      *
986      * @param payload the processed object
987      */
988     public void printPayload(@Nullable IBaseDataObject payload) {
989         if (payload == null) {
990             return;
991         }
992         outStream.println(payload.getFilename());
993         outStream.println("Current form: " + payload.getAllCurrentForms());
994         outStream.println("File type: " + payload.getFileType());
995         outStream.println("Encoding: " + payload.getFontEncoding());
996         outStream.println("Length: " + payload.dataLength());
997         if (payload.getNumAlternateViews() > 0) {
998             outStream.println("Alt views: " + payload.getAlternateViewNames());
999         }
1000         if (payload.getNumChildren() > 0) {
1001             outStream.println("Attachments: " + payload.getNumChildren());
1002         }
1003 
1004         if (metaPatterns != null && metaPatterns.length > 0) {
1005             for (Map.Entry<String, String> entry : payload.getCookedParameters().entrySet()) {

1007                     if (Pattern.matches(metaPatterns[i], entry.getKey())) {
1008                         outStream.println("Metadata " + entry.getKey() + ": " + entry.getValue());
1009                     }
1010                 }
1011             }
1012         }
1013 
1014         boolean needTrailingCr = true;
1015 
1016         if (payload.dataLength() > 0 && (isVerbose() || (viewsToPrint.contains("MAIN") && payload.getFilename().indexOf(Family.SEP) == -1))) {
1017             outStream.println("Data: <<EODATA");
1018             outStream.write(payload.data(), 0, payload.dataLength());
1019             outStream.println();
1020             outStream.println("EODATA");
1021             outStream.println();
1022             needTrailingCr = false;
1023         }
1024 
1025         for (String view : viewsToPrint) {
1026             byte[] av = payload.getAlternateView(view);
1027             if (av != null) {
1028                 outStream.println("Alternate View " + view);
1029                 outStream.write(av, 0, av.length);
1030                 outStream.println();
1031                 outStream.println();
1032                 needTrailingCr = false;
1033             }
1034         }
1035 
1036         if (needTrailingCr) {
1037             outStream.println();
1038         }
1039     }

1041     /**
1042      * Split the output to separate files. This is for doing follow on processing on the data, so just the data gets output
1043      * here. One payload or attachment per file. Directories will be created as needed underneath the baseOutputDir (can be
1044      * specified with -d).
1045      * <p>
1046      * Note: Output will be suppressed if an IBDO form or filename would yield an output path outside the configured
1047      * {@link #getBaseOutputDir() base output directory} path
1048      * </p>
1049      *
1050      * @param payload the processed payload
1051      * @param att the list of attachments
1052      */
1053     public void handleSplitOutput(IBaseDataObject payload, List<IBaseDataObject> att) {
1054         String fn = getBaseOutputDir() + "/" + payload.shortName() + "." + payload.currentForm();
1055 
1056         String safePath;
1057         try {
1058             safePath = filePathIsWithinBaseDirectory(getBaseOutputDir(), fn);
1059             if (Executrix.writeDataToFile(payload.data(), safePath)) {
1060                 logger.debug("Wrote output to {}", safePath);
1061             } else {
1062                 logger.error("Could not write output to {}", safePath);
1063             }
1064         } catch (IllegalArgumentException e) {
1065             logger.error("Could not write output to {}", fn, e);
1066         }
1067         for (IBaseDataObject part : att) {
1068             fn = getBaseOutputDir() + "/" + part.shortName() + "." + part.currentForm();
1069             try {
1070                 safePath = filePathIsWithinBaseDirectory(getBaseOutputDir(), fn);
1071                 if (Executrix.writeDataToFile(payload.data(), safePath)) {
1072                     logger.debug("Wrote attachment output to {}", safePath);
1073                 } else {
1074                     logger.error("Could not write output to {}", safePath);
1075                 }
1076             } catch (IllegalArgumentException e) {
1077                 logger.error("Could not write output to {}", fn, e);
1078             }
1079         }
1080     }

1082     /**
1083      * Normalizes the provided paths and tests to see whether the file path is within the required base directory.
1084      *
1085      * @param requiredBase required base directory
1086      * @param filePath file path to be tested
1087      * @return the normalized file path
1088      * @throws IllegalArgumentException if the filePath contains illegal characters or is outside the required base
1089      *         directory
1090      */
1091     static String filePathIsWithinBaseDirectory(final String requiredBase, final String filePath) throws IllegalArgumentException {
1092 
1093         if (StringUtils.isBlank(requiredBase)) {
1094             throw new IllegalArgumentException("requiredBase must not be blank");
1095         }
1096 
1097         if (StringUtils.isBlank(filePath)) {
1098             throw new IllegalArgumentException("filePath must not be blank");
1099         }
1100 
1101         // probably an overly simplistic test
1102         if (filePath.contains("..")) {
1103             throw new IllegalArgumentException("filePath contains illegal character sequence \"..\"");
1104         }
1105         Path normalizedBasePath = Paths.get(requiredBase).normalize().toAbsolutePath();
1106         Path normalizedFilePath = Paths.get(filePath).normalize().toAbsolutePath();
1107 
1108         // append path separator to defeat traversal via a sibling directory with a similar name
1109         if (!normalizedFilePath.startsWith(normalizedBasePath.toString() + "/")) {
1110             throw new IllegalArgumentException("Normalized file path (\"" + filePath + "\") is outside the required base path (\""
1111                     + requiredBase + "\")");
1112         }
1113         return normalizedFilePath.toString();
1114     }

1116     /**
1117      * Process an IBaseDataObject from teh work queue
1118      */
1119     protected class Worker implements Runnable {
1120         boolean idle = true;
1121         boolean timeToStop = false;
1122 
1123         public boolean isIdle() {
1124             return idle;
1125         }
1126 
1127         public void stop() {
1128             timeToStop = true;
1129         }
1130 
1131         /**
1132          * From the Runnable interface
1133          */
1134         @Override
1135         public void run() {
1136             IBaseDataObject payload = null;
1137 
1138             while (!timeToStop) {
1139                 // Try to get a payload
1140                 synchronized (workQueue) {
1141                     try {
1142                         payload = workQueue.removeFirst();
1143                     } catch (NoSuchElementException ignore) {
1144                         // empty catch block
1145                     }
1146                 }
1147 
1148                 // Process a payload if there is one
1149                 if (payload != null) {
1150                     idle = false;
1151                     processPayload(payload);
1152                     payload = null;
1153                     idle = true;
1154                 }
1155 
1156                 // Wait for a payload
1157                 else if (!timeToStop) {
1158                     synchronized (workQueue) {
1159                         try {
1160                             workQueue.wait(1000);
1161                         } catch (InterruptedException ignore) {
1162                             Thread.currentThread().interrupt();
1163                         }
1164                     }
1165                 }
1166             }
1167         }
1168     }
