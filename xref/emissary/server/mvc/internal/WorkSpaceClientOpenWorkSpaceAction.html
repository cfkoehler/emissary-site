24  @Path("")
25  // context is /emissary, set in EmissaryServer
26  public class WorkSpaceClientOpenWorkSpaceAction {
27      private final Logger logger = LoggerFactory.getLogger(this.getClass());
28  
29      /*
30       * <!-- Notify a PickUpPlaceClient of a new WorkSpace --> <Use-Case source="*"
31       * action="/WorkSpaceClientOpenWorkSpace.action"> <Work type="Bean" target="emissary.comms.http.worker.LogWorker"/>
32       * <Work type="Bean" target="emissary.comms.http.worker.WorkSpaceClientOpenSpaceWorker"/> <View status="0"
33       * view="/success.jsp"/> <View status="-1" view="/error.jsp"/> </Use-Case>
34       */
35  
36      @POST
37      @Path("/WorkSpaceClientOpenWorkSpace.action")
38      @Consumes(MediaType.APPLICATION_FORM_URLENCODED)
39      @Produces(MediaType.TEXT_PLAIN)
40      public Response openSpaceWorker(@FormParam(CLIENT_NAME) String placeName, @FormParam(SPACE_NAME) String spaceName) {
41          try {
42              return openTheSpace(placeName, spaceName);
43          } catch (IllegalArgumentException | NamespaceException e) {
44              logger.error("Problem opening the WorkSpace: {} - {}, {} - {}", CLIENT_NAME, placeName, SPACE_NAME, spaceName, e);
45              return Response.serverError().entity("There was a problem opening the workspace: " + e.getMessage()).build();
46          }
47      }
48  
49      private Response openTheSpace(String placeName, String spaceName) throws IllegalArgumentException, NamespaceException {
50          // TODO Figure out why we have to remove the key prefix now
51          // INITIAL.FILE_PICK_UP_CLIENT.INPUT.http://localhost:8001/FilePickUpClient
52          String placeKey = placeName.substring("INITIAL.FILE_PICK_UP_CLIENT.INPUT.".length());
53          final IPickUpSpace/../../emissary/pickup/IPickUpSpace.html#IPickUpSpace">IPickUpSpace place = (IPickUpSpace) Namespace.lookup(placeKey);
54          if (place == null) {
55              throw new IllegalArgumentException("No client place found using name " + placeName);
56          }
57  
58          logger.debug("Notified {} to open space at {}", placeName, spaceName);
59          place.openSpace(spaceName);
60          return Response.ok().entity("Successfully opened the workspace").build();
61      }
