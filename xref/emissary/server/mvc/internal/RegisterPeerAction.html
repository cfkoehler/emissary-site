44      @POST
45      @Path("/RegisterPeer.action")
46      @Consumes(MediaType.APPLICATION_FORM_URLENCODED)
47      @Produces(MediaType.APPLICATION_XML)
48      public Response registerPeerPost(@FormParam(DIRECTORY_NAME) String directoryName, @FormParam(TARGET_DIRECTORY) String targetDirectory) {
49          if (StringUtils.isBlank(directoryName) || StringUtils.isBlank(targetDirectory)) {
50              return Response.serverError().entity(HtmlEscaper.escapeHtml(
51                      "Bad Params: " + DIRECTORY_NAME + " - " + directoryName + ", " + TARGET_DIRECTORY + " - " + targetDirectory)).build();
52          }
53          return processRegisterPeer(directoryName, targetDirectory);
54      }
55  
56      private Response processRegisterPeer(String peerKey, String dirName) {
57          final IRemoteDirectory dir = new IRemoteDirectory.Lookup().getLocalDirectory(dirName);
58          if (dir == null) {
59              // If we get here, there was a problem looking up the IRemoteDirectory
60              // old RegisterPlaceWorker failure
61              // return new WorkerStatus(WorkerStatus.FAILURE, "Could not register peer " + peerKey);
62              return Response.serverError().entity(
63                      "Remote directory lookup failed for dirName: " + HtmlEscaper.escapeHtml(dirName)).build();
64          }
65  
66          logger.info("Attempting to register peer {} with {}", peerKey, dirName);
67  
68          try {
69              final DirectoryPlace../../../emissary/directory/DirectoryPlace.html#DirectoryPlace">DirectoryPlace d = (DirectoryPlace) Namespace.lookup(dirName);
70              if (!d.isStaticPeer(KeyManipulator.getDefaultDirectoryKey(peerKey))) {
71                  logger.warn("Contact attempted from {} but it is not a configured peer", peerKey);
72                  return Response.serverError().entity("Registration failed for peer: " + HtmlEscaper.escapeHtml(peerKey)).build();
73              }
74          } catch (NamespaceException e) {
75              logger.warn("Problem performing namespace lookup for {}", dirName);
76              return Response.serverError().entity("Registration failed for peer: " + HtmlEscaper.escapeHtml(peerKey)).build();
77          }
78  
79          final Set<String> set = new HashSet<>();
80          set.add(KeyManipulator.getDefaultDirectoryKey("#" + peerKey));
81          MDC.put(MDCConstants.SERVICE_LOCATION, KeyManipulator.getServiceLocation(dir.getKey()));
82          try {
83              dir.irdAddPeerDirectories(set);
84          } finally {
85              MDC.remove(MDCConstants.SERVICE_LOCATION);
86          }
87  
88          logger.info("Registered peer {}", peerKey);
89  
90          return Response.ok().entity(DirectoryXmlContainer.toXmlString(dir)).build();
91      }
