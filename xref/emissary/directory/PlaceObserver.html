3   /**
4    * This interface can be implemented by any class wishing to subscribe to place changes in an emissary DirectoryPlace.
5    * These methods are called for the appropriate actions that match the pattern supplied to the observable.
6    */
7   public interface PlaceObserver extends DirectoryObserver {
8       /**
9        * Called when a place matching the subscription is registered
10       * 
11       * @param observableKey key of the directory reporting the registration
12       * @param placeKey key of the place that is being registered
13       */
14      void placeRegistered(String observableKey, String placeKey);
15  
16      /**
17       * Called when a place matching the subscription is deregistered
18       * 
19       * @param observableKey key of the directory reporting the deregistration
20       * @param placeKey key of the place that is being deregistered
21       */
22      void placeDeregistered(String observableKey, String placeKey);
23  
24      /**
25       * Called when the cost of a place matching the subscription is changed
26       * 
27       * @param observableKey key of the directory reporting the change
28       * @param placeKey key of the place that is being changed
29       */
30      void placeCostChanged(String observableKey, String placeKey);
31  
32      /**
33       * The pattern for this observers subscription,
34       * 
35       * @see emissary.directory.KeyManipulator#gmatch
36       */
37      String getPattern();
38  }
