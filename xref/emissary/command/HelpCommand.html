1   package emissary.command;
2   
3   import com.beust.jcommander.JCommander;
4   import com.beust.jcommander.Parameter;
5   import com.beust.jcommander.ParameterException;
6   import com.beust.jcommander.Parameters;
7   import org.slf4j.Logger;
8   import org.slf4j.LoggerFactory;
9   
10  import java.util.ArrayList;
11  import java.util.List;
12  import java.util.Map.Entry;
13  
14  @Parameters(commandDescription = "Print commands or usage for subcommand")
15  public class HelpCommand implements EmissaryCommand {
16  
17      static final Logger LOG = LoggerFactory.getLogger(HelpCommand.class);
18  
19      public static String COMMAND_NAME = "help";
20  
21      @Parameter(arity = 1, description = "display usage for this subcommand ")
22      public List<String> subcommands = new ArrayList<>();
23  
24  
25      public List<String> getSubcommands() {
26          return subcommands;
27      }
28  
29      public String getSubcommand() {
30          return subcommands.get(0);
31      }
32  
33      @Override
34      public String getCommandName() {
35          return COMMAND_NAME;
36      }
37  
38      @Override
39      public void run(JCommander jc) {
40          setup();
41          if (subcommands.size() == 0) {
42              dumpCommands(jc);
43          } else if (subcommands.size() > 1) {
44              LOG.error("You can only see help for 1 command at a time");
45              dumpCommands(jc);
46          } else {
47              String subcommand = getSubcommand();
48              LOG.info("Detailed help for: " + subcommand);
49              try {
50                  jc.usage(subcommand);
51              } catch (ParameterException e) {
52                  LOG.error("ERROR: invalid command name: " + subcommand);
53                  dumpCommands(jc);
54              }
55          }
56      }
57  
58      @Override
59      public void setupCommand() {
60          // nothing for this command
61      }
62  
63      public static void dumpCommands(JCommander jc) {
64          System.out.println("Available commands:");
65          for (Entry<String, JCommander> cmd : jc.getCommands().entrySet()) {
66              String name = cmd.getKey();
67              String description = jc.getCommandDescription(name);
68              LOG.info("\t" + String.format("%1$-15s", name) + " " + description);
69          }
70          LOG.info("Use 'help <command-name>' to see more detailed info about that command");
71      }
72  
73      @Override
74      public void outputBanner() {
75          new Banner().dump();
76      }
77  
78  }
