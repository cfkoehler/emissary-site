9   import static emissary.server.api.Directories.DIRECTORIES_ENDPOINT;
10  
11  @Command(description = "List all of the active directories", subcommands = {HelpCommand.class})
12  public class DirectoryCommand extends MonitorCommand<DirectoryResponseEntity> {
13  
14      public static final String COMMAND_NAME = "directory";
15      public static final int DEFAULT_PORT = 8001;
16  
17      @Override
18      public int getDefaultPort() {
19          return DEFAULT_PORT;
20      }
21  
22      @Override
23      public String getCommandName() {
24          return COMMAND_NAME;
25      }
26  
27      @Override
28      public DirectoryResponseEntity sendRequest(EmissaryClient client, String endpoint) {
29          return client.send(new HttpGet(endpoint)).getContent(DirectoryResponseEntity.class);
30      }
31  
32      @Override
33      public String getTargetEndpoint() {
34          return DIRECTORIES_ENDPOINT;
35      }
36  }
