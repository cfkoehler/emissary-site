1   package emissary.command.converter;
2   
3   import emissary.pickup.Priority;
4   import emissary.pickup.PriorityDirectory;
5   
6   import com.beust.jcommander.IStringConverter;
7   
8   public class PriorityDirectoryConverter implements IStringConverter<PriorityDirectory> {
9       public static final String PRIORITY_DIR_REGEX = ".*:\\d+$";
10  
11      @Override
12      public PriorityDirectory convert(String value) {
13          // Take from the old WorkSpace.java class
14          if (value.matches(PRIORITY_DIR_REGEX)) {
15              final int pos = value.lastIndexOf(":");
16              String dirName = value.substring(0, pos);
17              if (!dirName.endsWith("/")) {
18                  dirName += "/";
19              }
20              final int priority = Integer.parseInt(value.substring(pos + 1));
21              return new PriorityDirectory(dirName, priority);
22          } else {
23              if (!value.endsWith("/")) {
24                  value += "/";
25              }
26              return new PriorityDirectory(value, Priority.DEFAULT);
27          }
28      }
29  }
